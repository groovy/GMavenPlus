137     /**
138      * @param fileSet
139      * @return
140      */
141     protected List<String> getSources(FileSet fileSet) {
142         List<String> files = new ArrayList<String>();
143         FileSetManager fileSetManager = new FileSetManager(getLog());
144 
145         if (fileSet != null) {
146             for (String include : Arrays.asList(fileSetManager.getIncludedFiles(fileSet))) {
147                 files.add(include);
148             }
149         } else {
150             FileSet fs = new FileSet();
151             String directory = project.getBasedir().getAbsolutePath() + File.separator + "src" + File.separator + "main" + File.separator + "groovy";
152             fs.setDirectory(directory);
153             fs.setIncludes(Arrays.asList(DEFAULT_SOURCE_PATTERN));
154             String[] includes = fileSetManager.getIncludedFiles(fs);
155             for (String file : includes) {
156                 files.add(directory + File.separator + file);
157             }
158         }
159 
160         return files;
161     }

163     /**
164      * @param sourceDirectories
165      * @param outputDirectory
166      * @throws ClassNotFoundException
167      * @throws InvocationTargetException
168      * @throws IllegalAccessException
169      * @throws InstantiationException
170      */
171     protected void generateGroovydoc(FileSet[] sourceDirectories, File outputDirectory) throws ClassNotFoundException, InvocationTargetException, IllegalAccessException, InstantiationException {
172         // get classes we need with reflection
173         Class<?> groovyDocToolClass = Class.forName("org.codehaus.groovy.tools.groovydoc.GroovyDocTool");
174         Class<?> outputToolClass = Class.forName("org.codehaus.groovy.tools.groovydoc.OutputTool");
175         Class<?> fileOutputToolClass = Class.forName("org.codehaus.groovy.tools.groovydoc.FileOutputTool");
176         Class<?> resourceManagerClass = Class.forName("org.codehaus.groovy.tools.groovydoc.ResourceManager");
177         Class<?> classpathResourceManagerClass = Class.forName("org.codehaus.groovy.tools.groovydoc.ClasspathResourceManager");
178 
179         // set up Groovydoc options
180         List links = new ArrayList();
181         Properties properties = new Properties();
182         properties.setProperty("windowTitle", windowTitle);
183         properties.setProperty("docTitle", docTitle);
184         properties.setProperty("footer", footer);
185         properties.setProperty("header", header);
186         properties.setProperty("author", Boolean.toString(displayAuthor));
187         properties.setProperty("overviewFile", overviewFile != null ? overviewFile.getAbsolutePath() : "");
188         try {
189             Scopes scopeVal = Scopes.valueOf(scope.toUpperCase());
190             switch (scopeVal) {
191                 case PUBLIC:
192                     properties.setProperty("publicScope", "true");
193                     break;
194                 case PROTECTED:
195                     properties.setProperty("protectedScope", "true");
196                     break;
197                 case PACKAGE:
198                     properties.setProperty("packageScope", "true");
199                     break;
200                 case PRIVATE:
201                     properties.setProperty("privateScope", "true");
202                     break;
203             }
204         } catch (IllegalArgumentException e) {
205             getLog().warn("Scope (" + scope + ") was not recognized.  Skipping argument.");
206         }
207         Object fileOutputTool = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(fileOutputToolClass));
208         Object classpathResourceManager = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(classpathResourceManagerClass));
209 
210         // generate Groovydoc
211         for (FileSet sourceDirectory : sourceDirectories) {
212             Object groovyDocTool;
213             groovyDocTool = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyDocToolClass, resourceManagerClass, String[].class, String[].class, String[].class, String[].class, List.class, Properties.class),
214                     classpathResourceManager,
215                     new String[] {sourceDirectory.getDirectory()},
216                     GroovyDocTemplateInfo.DEFAULT_DOC_TEMPLATES,
217                     GroovyDocTemplateInfo.DEFAULT_PACKAGE_TEMPLATES,
218                     GroovyDocTemplateInfo.DEFAULT_CLASS_TEMPLATES,
219                     links,
220                     properties
221             );
222             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyDocToolClass, "add", List.class), groovyDocTool, getSources(sourceDirectory));
223             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyDocToolClass, "renderToOutput", outputToolClass, String.class), groovyDocTool, fileOutputTool, outputDirectory.getAbsolutePath());
224         }
225 
226         // overwrite stylesheet.css with provided stylesheet (if configured)
227         if (stylesheetFile != null) {
228             copyStylesheet(outputDirectory);
229         }
230     }
231 
232     private void copyStylesheet(File outputDirectory) {
233         getLog().info("Using stylesheet from " + stylesheetFile.getAbsolutePath() + ".");
234         try {
235             BufferedReader in = new BufferedReader(new InputStreamReader(new FileInputStream(stylesheetFile), stylesheetEncoding));
236             StringBuilder css = new StringBuilder();
237             String line;
238             while ((line = in.readLine()) != null) {
239                 css.append(line).append("\n");
240             }
241             in.close();
242             File outfile = new File(outputDirectory, "stylesheet.css");
243             BufferedWriter out = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(outfile), stylesheetEncoding));
244             out.write(css.toString());
245             out.flush();
246             out.close();
247         } catch (IOException e) {
248             getLog().warn("Unable to copy specified stylesheet (" + stylesheetFile.getAbsolutePath() + ").", e);
249         }
250     }
