116     /**
117      * Gets the version string of Groovy used from the dependency information.
118      *
119      * @return The version string of Groovy used by the project
120      */
121     protected String getGroovyVersionString() {
122         String groovyVersion = null;
123 
124         /*
125          * You can call InvokerHelper.getVersion() for versions 1.0 - 1.8.x but
126          * not for 1.9+.
127          * You can call GroovySystem.getVersion() for versions 1.6.6+.
128          * And for some reason InvokerHelper.getVersion() was returning an empty
129          * String for 1.5.0, so I decided to just get it from the dependency itself.
130          */
131         Artifact groovyDependency = getGroovyDependency();
132 
133         if (groovyDependency == null) {
134             getLog().error("Unable to determine Groovy version.  Is Groovy declared as a dependency?");
135         } else {
136             groovyVersion = groovyDependency.getVersion();
137         }
138 
139         return groovyVersion;
140     }

151     /**
152      * Gets the version of Groovy used from the dependency information.
153      *
154      * @return <code>true</code> if the version of Groovy uses InvokeDynamic,
155      *         <code>false</code> if not or Groovy dependency cannot be found.
156      */
157     protected boolean isGroovyIndy() {
158         boolean isGroovyIndy = false;
159 
160         Artifact groovyDependency = getGroovyDependency();
161         if (groovyDependency == null) {
162             getLog().error("Unable to determine Groovy version.  Is Groovy declared as a dependency?");
163         } else if ("indy".equals(groovyDependency.getClassifier())) {
164             isGroovyIndy = true;
165         }
166 
167         return isGroovyIndy;
168     }

170     /**
171      * Gets the Groovy dependency used by the project.
172      *
173      * @return The Groovy dependency used by the project
174      */
175     protected Artifact getGroovyDependency() {
176         Artifact groovyDependency = null;
177 
178         if (pluginArtifacts != null) {
179             for (Object art : pluginArtifacts) {
180                 Artifact artifact = (Artifact) art;
181                 if (isGroovyGroupId(artifact) && isGroovyArtifactId(artifact) && artifact.getType().equals("jar")) {
182                     groovyDependency = artifact;
183                     break;
184                 }
185             }
186         }
187 
188         if (groovyDependency == null && project.getCompileDependencies() != null) {
189             for (Object dep : project.getCompileDependencies()) {
190                 Dependency dependency = (Dependency) dep;
191                 if (isGroovyGroupId(dependency) && isGroovyArtifactId(dependency) && dependency.getType().equals("jar")) {
192                     groovyDependency = new DefaultArtifact(dependency.getGroupId(), dependency.getArtifactId(), VersionRange.createFromVersion(dependency.getVersion()), dependency.getScope(), dependency.getType(), dependency.getClassifier() != null ? dependency.getClassifier() : "", null);
193                     break;
194                 }
195             }
196         }
197 
198         return groovyDependency;
199     }

201     /**

208         return dependency.getGroupId().equals("org.codehaus.groovy") || dependency.getGroupId().equals("groovy");

220                                     || dependency.getArtifactId().equals("groovy-jdk14");
