17  package org.codehaus.gmavenplus.mojo;
18  
19  import org.apache.maven.model.Dependency;
20  import org.apache.maven.plugin.AbstractMojo;
21  import org.apache.maven.project.MavenProject;
22  
23  import java.io.File;
24  
25  
26  /**
27   * The base mojo class, which all other mojos extend.
28   *
29   * @author Keegan Witt
30   *
31   * @requiresDependencyResolution compile
32   * @configurator include-project-dependencies
33   */
34  public abstract class AbstractGroovyMojo extends AbstractMojo {
35  
36      /**
37       * The pattern defining Groovy files.
38       */
39      protected static final String GROOVY_SOURCES_PATTERN = "**" + File.separator + "*.groovy";
40  
41      /**
42       * The pattern defining Java stub files.
43       */
44      protected static final String JAVA_SOURCES_PATTERN = "**" + File.separator + "*.java";
45  
46      /**
47       * The Maven project this plugin is being used on.
48       *
49       * @parameter property="project"
50       * @required
51       * @readonly
52       */
53      protected MavenProject project;
54  
55      /**
56       * Logs the version of groovy used by this mojo.
57       *
58       * @param goal The goal to mention in the log statement showing Groovy version
59       */
60      protected void logGroovyVersion(final String goal) {
61          if (getLog().isInfoEnabled()) {
62              getLog().info("Using Groovy " + getGroovyVersion() + " from project compile classpath to perform " + goal + ".");
63          }
64      }
65  
66      /**
67       * Gets the version of Groovy used from the dependency information.
68       *
69       * @return The version Groovy used by the project
70       */
71      protected String getGroovyVersion() {
72          String groovyVersion = null;
73  
74          /*
75           * You can call InvokerHelper.getVersion() for versions 1.0 - 1.8.x but
76           * not for 1.9+.
77           * You can call GroovySystem.getVersion() for versions 1.6.6+.
78           * And for some reason InvokerHelper.getVersion() was returning an empty
79           * String for 1.5.0, so I decided to just get it from the dependency itself.
80           */
81          Dependency groovyDependency = getGroovyDependency();
82  
83          if (groovyDependency == null) {
84              getLog().error("Unable to determine Groovy version.");
85          } else {
86              groovyVersion = groovyDependency.getVersion();
87          }
88  
89          return groovyVersion;
90      }
91  
92      /**
93       * Gets the version of Groovy used from the dependency information.
94       *
95       * @return <code>true</code> if the version of Groovy uses InvokeDynamic,
96       *         <code>false</code> if not or Groovy dependency cannot be found.
97       */
98      protected boolean isGroovyIndy() {
99          boolean isGroovyIndy = false;
100 
101         Dependency groovyDependency = getGroovyDependency();
102         if (groovyDependency == null) {
103             getLog().error("Unable to determine Groovy version.");
104         } else if ("indy".equals(groovyDependency.getClassifier())) {
105             isGroovyIndy = true;
106         }
107 
108         return isGroovyIndy;
109     }
110 
111     /**
112      * Gets the Groovy dependency used by the project.
113      *
114      * @return The Groovy dependency used by the project
115      */
116     protected Dependency getGroovyDependency() {
117         Dependency groovyDependency = null;
118 
119         for (Object dep : project.getCompileDependencies()) {
120             Dependency dependency = (Dependency) dep;
121             if (isGroovyGroupId(dependency) && isGroovyArtifactId(dependency) && dependency.getType().equals("jar")) {
122                 groovyDependency = dependency;
123                 break;
124             }
125         }
126 
127         return groovyDependency;
128     }
129 
130     /**
131      * Whether the groupId of the dependency is Groovy's groupId.
132      *
133      * @param dependency The dependency to inspect
134      * @return <code>true</code> if the dependency's groupId is a Groovy groupId, <code>false</code> otherwise
135      */
136     protected boolean isGroovyGroupId(final Dependency dependency) {
137         return dependency.getGroupId().equals("org.codehaus.groovy") || dependency.getGroupId().equals("groovy");
138     }
139 
140     /**
141      * Whether the artifactId of the dependency is Groovy's artifactId.
142      *
143      * @param dependency The dependency to inspect
144      * @return <code>true</code> if the dependency's groupId is a Groovy groupId, <code>false</code> otherwise
145      */
146     protected boolean isGroovyArtifactId(final Dependency dependency) {
147         return dependency.getArtifactId().equals("groovy-all") || dependency.getArtifactId().equals("groovy-all-minimal")
148                                     || dependency.getArtifactId().equals("groovy") || dependency.getArtifactId().equals("groovy-all-jdk14")
149                                     || dependency.getArtifactId().equals("groovy-jdk14");
150     }
151 
152 }
