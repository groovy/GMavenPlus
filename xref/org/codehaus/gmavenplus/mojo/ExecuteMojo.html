17  package org.codehaus.gmavenplus.mojo;
18  
19  import com.google.common.io.Closer;
20  import org.codehaus.gmavenplus.model.Version;
21  import org.codehaus.gmavenplus.util.ReflectionUtils;
22  import org.apache.maven.plugin.MojoExecutionException;
23  import org.apache.maven.plugin.MojoFailureException;
24  import java.io.BufferedReader;
25  import java.io.IOException;
26  import java.io.InputStreamReader;
27  import java.lang.reflect.InvocationTargetException;
28  import java.net.MalformedURLException;
29  import java.net.URL;
30  import java.util.Properties;
31  
32  
33  /**
34   * Executes Groovy scripts (in the pom or external).
35   *
36   * @author Keegan Witt
37   *
38   * @goal execute
39   * @configurator include-project-test-dependencies
40   * @requiresDependencyResolution test
41   * @threadSafe
42   */
43  public class ExecuteMojo extends AbstractGroovyMojo {
44  
45      /**
46       * The minimum version of Groovy that this mojo supports.
47       */
48      protected static final Version MIN_GROOVY_VERSION = new Version(1, 5, 0);
49  
50      /**
51       * Groovy scripts to run (in order).  Can be an actual Groovy script or a
52       * {@link java.net.URL URL} to a Groovy script (local or remote).
53       *
54       * @parameter
55       * @required
56       */
57      protected String[] scripts;
58  
59      /**
60       * Whether to continue executing remaining scripts when a script fails.
61       *
62       * @parameter default-value="false"
63       */
64      protected boolean continueExecuting;
65  
66      /**
67       * The encoding of script files.
68       *
69       * @parameter default-value="${project.build.sourceEncoding}"
70       */
71      protected String sourceEncoding;
72  
73      /**
74       * Properties to make available in scripts.  By default it will include
75       * <dl>
76       *   <dt>settings</dt>
77       *     <dd>a org.apache.maven.settings.Settings object of the current Maven settings</dd>
78       *   <dt>project</dt>
79       *     <dd>a org.apache.maven.project.Project object of the current Maven project</dd>
80       *   <dt>session</dt>
81       *     <dd>a org.apache.maven.execution.MavenSession object of the current Maven session</dd>
82       *   <dt>pluginArtifacts</dt>
83       *     <dd>a list of org.apache.maven.artifact.Artifact objects of this plugin's artifacts</dd>
84       *   <dt>localRepository</dt>
85       *     <dd>a org.apache.maven.artifact.repository.ArtifactRepository object of Maven's local repository</dd>
86       *   <dt>reactorProjects</dt>
87       *     <dd>a list of org.apache.maven.project.MavenProject objects currently loaded by the reactor</dd>
88       * </dl>
89       * These can be overridden.
90       *
91       * @parameter
92       */
93      protected Properties properties = new Properties();
94  
95      /**
96       * Executes this mojo.
97       *
98       * @throws MojoExecutionException If an unexpected problem occurs. Throwing this exception causes a "BUILD ERROR" message to be displayed
99       * @throws MojoFailureException If an expected problem (such as a compilation failure) occurs. Throwing this exception causes a "BUILD FAILURE" message to be displayed
100      */
101     public void execute() throws MojoExecutionException, MojoFailureException {
102         if (groovyVersionSupportsAction()) {
103             logGroovyVersion("execute");
104             if (scripts == null || scripts.length == 0) {
105                 getLog().info("No scripts specified for execution.  Skipping.");
106                 return;
107             }
108 
109             try {
110                 // get classes we need with reflection
111                 Class<?> groovyShellClass = Class.forName("groovy.lang.GroovyShell");
112 
113                 // create a GroovyShell to run scripts in
114                 Object shell = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyShellClass));
115                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "setProperty", String.class, Object.class), shell, "settings", settings);
116                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "setProperty", String.class, Object.class), shell, "project", project);
117                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "setProperty", String.class, Object.class), shell, "session", session);
118                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "setProperty", String.class, Object.class), shell, "pluginArtifacts", pluginArtifacts);
119                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "setProperty", String.class, Object.class), shell, "localRepository", localRepository);
120                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "setProperty", String.class, Object.class), shell, "reactorProjects", reactorProjects);
121                 // this is intentionally after the default properties so that the user can override if desired
122                 for (String key : properties.stringPropertyNames()) {
123                     ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "setProperty", String.class, Object.class), shell, key, properties.getProperty(key));
124                 }
125 
126                 // TODO: load configurable (compile, test, runtime, or system) dependencies onto classpath before executing so they can be used in scripts?
127 
128                 // run the scripts
129                 int scriptNum = 1;
130                 for (String script : scripts) {
131                     Closer closer = Closer.create();
132                     try {
133                         try {
134                             URL url = new URL(script);
135                             // it's a URL to a script
136                             getLog().info("Fetching Groovy script from " + url.toString() + ".");
137                             BufferedReader reader = closer.register(new BufferedReader(new InputStreamReader(url.openStream(), sourceEncoding)));
138                             StringBuilder scriptSource = new StringBuilder();
139                             String line;
140                             while ((line = reader.readLine()) != null) {
141                                 scriptSource.append(line).append("\n");
142                             }
143                             if (!scriptSource.toString().isEmpty()) {
144                                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "evaluate", String.class), shell, scriptSource.toString());
145                             }
146                         } catch (MalformedURLException e) {
147                             // it's not a URL to a script, treat as a script body
148                             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "evaluate", String.class), shell, script);
149                         } catch (Throwable throwable) {
150                             throw closer.rethrow(throwable);
151                         } finally {
152                             closer.close();
153                         }
154                     } catch (IOException ioe) {
155                         if (continueExecuting) {
156                             getLog().error("An Exception occurred while executing script " + scriptNum + ".  Continuing to execute remaining scripts.", ioe);
157                         } else {
158                             throw new MojoExecutionException("An Exception occurred while executing script " + scriptNum + ".", ioe);
159                         }
160                     }
161                     scriptNum++;
162                 }
163             } catch (ClassNotFoundException e) {
164                 throw new MojoExecutionException("Unable to get a Groovy class from classpath.  Do you have Groovy as a compile dependency in your project?", e);
165             } catch (InvocationTargetException e) {
166                 throw new MojoExecutionException("Error occurred while calling a method on a Groovy class from classpath.", e);
167             } catch (InstantiationException e) {
168                 throw new MojoExecutionException("Error occurred while instantiating a Groovy class from classpath.", e);
169             } catch (IllegalAccessException e) {
170                 throw new MojoExecutionException("Unable to access a method on a Groovy class from classpath.", e);
171             }
172         } else {
173             getLog().error("Your Groovy version (" + getGroovyVersion() + ") script execution.  The minimum version of Groovy required is " + MIN_GROOVY_VERSION + ".  Skipping script execution.");
174         }
175     }
176 
177     /**
178      * Determines whether this mojo can be run with the version of Groovy supplied.
179      * Must be >= 1.8.2 because not all the classes needed were available and
180      * functioning correctly in previous versions.
181      *
182      * @return <code>true</code> only if the version of Groovy supports this mojo.
183      */
184     protected boolean groovyVersionSupportsAction() {
185         return getGroovyVersion() != null && getGroovyVersion().compareTo(MIN_GROOVY_VERSION) >= 0;
186     }
187 
188 }
