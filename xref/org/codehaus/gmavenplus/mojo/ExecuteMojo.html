17  package org.codehaus.gmavenplus.mojo;
18  
19  import com.google.common.io.Closer;
20  import org.codehaus.gmavenplus.util.ReflectionUtils;
21  import org.apache.maven.plugin.MojoExecutionException;
22  import org.apache.maven.plugin.MojoFailureException;
23  import java.io.BufferedReader;
24  import java.io.IOException;
25  import java.io.InputStreamReader;
26  import java.lang.reflect.InvocationTargetException;
27  import java.net.MalformedURLException;
28  import java.net.URL;
29  
30  
31  /**
32   * Executes Groovy scripts (in the pom or external).
33   *
34   * @author Keegan Witt
35   *
36   * @goal execute
37   */
38  public class ExecuteMojo extends AbstractGroovyMojo {
39  
40      /**
41       * Groovy scripts to run (in order).  Can be an actual Groovy script or a
42       * {@link java.net.URL URL} to a Groovy script (local or remote).
43       *
44       * @parameter
45       * @required
46       */
47      protected String[] scripts;
48  
49      /**
50       * Whether to continue executing remaining scripts when a script fails
51       *
52       * @parameter default-value="false"
53       */
54      protected boolean continueExecuting;
55  
56      /**
57       * Encoding of script files
58       *
59       * @parameter default-value="${project.build.sourceEncoding}"
60       */
61      protected String sourceEncoding;
62  
63      /**
64       * Executes this mojo.
65       *
66       * @throws MojoExecutionException
67       * @throws MojoFailureException
68       */
69      public void execute() throws MojoExecutionException, MojoFailureException {
70          logGroovyVersion("execute");
71  
72          try {
73              // get classes we need with reflection
74              Class<?> groovyShellClass = Class.forName("groovy.lang.GroovyShell");
75  
76              // create a GroovyShell to run scripts in
77              Object shell = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyShellClass));
78  
79              // TODO: load runtime project dependencies onto classpath before executing so they can be used in scripts?
80  
81              // run the scripts
82              int scriptNum = 1;
83              for (String script : scripts) {
84                  Closer closer = Closer.create();
85                  try {
86                      try {
87                          URL url = new URL(script);
88                          // it's a URL to a script
89                          getLog().info("Fetching Groovy script from " + url.toString() + ".");
90                          BufferedReader reader = closer.register(new BufferedReader(new InputStreamReader(url.openStream(), sourceEncoding)));
91                          StringBuilder scriptSource = new StringBuilder();
92                          String line;
93                          while ((line = reader.readLine()) != null) {
94                              scriptSource.append(line).append("\n");
95                          }
96                          if (!scriptSource.toString().isEmpty()) {
97                              ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "evaluate", String.class), shell, scriptSource.toString());
98                          }
99                      } catch (MalformedURLException e) {
100                         // it's not a URL to a script, treat as a script body
101                         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "evaluate", String.class), shell, script);
102                     } catch (Throwable throwable) {
103                         throw closer.rethrow(throwable);
104                     } finally {
105                         closer.close();
106                     }
107                 } catch (IOException ioe) {
108                     if (continueExecuting) {
109                         getLog().error("An Exception occurred while executing script " + scriptNum + ".  Continuing to execute remaining scripts.", ioe);
110                     } else {
111                         throw new MojoExecutionException("An Exception occurred while executing script " + scriptNum + ".", ioe);
112                     }
113                 }
114                 scriptNum++;
115             }
116         } catch (ClassNotFoundException e) {
117             throw new MojoExecutionException("Unable to get a Groovy class from classpath.  Do you have Groovy as a compile dependency in your project?", e);
118         } catch (InvocationTargetException e) {
119             throw new MojoExecutionException("Error occurred while calling a method on a Groovy class from classpath.", e);
120         } catch (InstantiationException e) {
121             throw new MojoExecutionException("Error occurred while instantiating a Groovy class from classpath.", e);
122         } catch (IllegalAccessException e) {
123             throw new MojoExecutionException("Unable to access a method on a Groovy class from classpath.", e);
124         }
125     }
