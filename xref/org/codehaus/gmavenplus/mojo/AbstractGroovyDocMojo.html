260     /**
261      * Generates the GroovyDoc for the specified sources.
262      *
263      * @param sourceDirectories The source directories to generate GroovyDoc for
264      * @param classpath         The classpath to use for compilation
265      * @param outputDirectory   The directory to save the generated GroovyDoc in
266      * @throws ClassNotFoundException    when a class needed for GroovyDoc generation cannot be found
267      * @throws InstantiationException    when a class needed for GroovyDoc generation cannot be instantiated
268      * @throws IllegalAccessException    when a method needed for GroovyDoc generation cannot be accessed
269      * @throws InvocationTargetException when a reflection invocation needed for GroovyDoc generation cannot be completed
270      * @throws MalformedURLException     when a classpath element provides a malformed URL
271      */
272     protected synchronized void doGroovyDocGeneration(final FileSet[] sourceDirectories, final List<?> classpath, final File outputDirectory) throws ClassNotFoundException, InvocationTargetException, IllegalAccessException, InstantiationException, MalformedURLException {
273         if (skipGroovyDoc) {
274             getLog().info("Skipping generation of GroovyDoc because ${skipGroovydoc} was set to true.");
275             return;
276         }
277 
278         if (sourceDirectories == null || sourceDirectories.length == 0) {
279             getLog().info("No source directories specified for GroovyDoc generation. Skipping.");
280             return;
281         }
282 
283         setupClassWrangler(classpath, includeClasspath);
284 
285         classWrangler.logGroovyVersion(mojoExecution.getMojoDescriptor().getGoal());
286         logPluginClasspath();
287 
288         if (!groovyVersionSupportsAction()) {
289             getLog().error("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support GroovyDoc. The minimum version of Groovy required is " + minGroovyVersion + ". Skipping GroovyDoc generation.");
290             return;
291         }
292         if (groovyIs(GROOVY_1_6_0_RC1) || groovyIs(GROOVY_1_5_8)) {
293             // Groovy 1.5.8 and 1.6-RC-1 are blacklisted because of their dependency on org.apache.tools.ant.types.Path in GroovyDocTool constructor
294             getLog().warn("Groovy " + GROOVY_1_5_8 + " and " + GROOVY_1_6_0_RC1 + " are blacklisted from the supported GroovyDoc versions because of their dependency on Ant. Skipping GroovyDoc generation.");
295             return;
296         }
297 
298         // get classes we need with reflection
299         Class<?> groovyDocToolClass = classWrangler.getClass(this.groovyDocToolClass == null ? "org.codehaus.groovy.tools.groovydoc.GroovyDocTool" : this.groovyDocToolClass);
300         Class<?> outputToolClass = classWrangler.getClass(this.outputToolClass == null ? "org.codehaus.groovy.tools.groovydoc.OutputTool" : this.outputToolClass);
301         Class<?> fileOutputToolClass = classWrangler.getClass(this.fileOutputToolClass == null ? "org.codehaus.groovy.tools.groovydoc.FileOutputTool" : this.fileOutputToolClass);
302         Class<?> resourceManagerClass = classWrangler.getClass(this.resourceManagerClass == null ? "org.codehaus.groovy.tools.groovydoc.ResourceManager" : this.resourceManagerClass);
303         Class<?> classpathResourceManagerClass = classWrangler.getClass(this.classpathResourceManagerClass == null ? "org.codehaus.groovy.tools.groovydoc.ClasspathResourceManager" : this.classpathResourceManagerClass);
304 
305         // set up GroovyDoc options
306         if (attachGroovyDocAnnotation) {
307             if (groovyAtLeast(GROOVY_3_0_0_ALPHA_4)) {
308                 System.setProperty("runtimeGroovydoc", "true");
309             } else {
310                 getLog().warn("Requested to enable attaching GroovyDoc annotation, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_3_0_0_ALPHA_4 + " or newer). Ignoring enableGroovyDocAnnotation parameter.");
311             }
312         }
313         Properties docProperties = setupProperties();
314         Object fileOutputTool = invokeConstructor(findConstructor(fileOutputToolClass));
315         Object classpathResourceManager = invokeConstructor(findConstructor(classpathResourceManagerClass));
316         FileSetManager fileSetManager = new FileSetManager();
317         List<String> sourceDirectoriesStrings = new ArrayList<>();
318         for (FileSet sourceDirectory : sourceDirectories) {
319             sourceDirectoriesStrings.add(sourceDirectory.getDirectory());
320         }
321         GroovyDocTemplateInfo groovyDocTemplateInfo = new GroovyDocTemplateInfo(classWrangler.getGroovyVersion());
322         List<?> groovyDocLinks = setupLinks();
323         if (groovyOlderThan(GROOVY_1_6_0_RC2)) {
324             getLog().warn("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support GroovyDoc documentation properties (docTitle, footer, header, displayAuthor, overviewFile, and scope). You need Groovy 1.6-RC-2 or newer to support this. Ignoring properties.");
325         }
326 
327         // prevent Java stubs (which lack Javadoc) from overwriting GroovyDoc by removing Java sources
328         List<String> groovyDocSources = setupGroovyDocSources(sourceDirectories, fileSetManager);
329 
330         // instantiate GroovyDocTool
331         Object groovyDocTool = createGroovyDocTool(groovyDocToolClass, resourceManagerClass, docProperties, classpathResourceManager, sourceDirectoriesStrings, groovyDocTemplateInfo, groovyDocLinks);
332 
333         // generate GroovyDoc
334         generateGroovyDoc(outputDirectory, groovyDocToolClass, outputToolClass, fileOutputTool, groovyDocSources, groovyDocTool);
335 
336         // overwrite stylesheet.css with provided stylesheet (if configured)
337         if (stylesheetFile != null) {
338             copyStylesheet(outputDirectory);
339         }
340     }

342     /**
343      * Sets up the documentation properties.
344      *
345      * @return the documentation properties
346      */
347     protected Properties setupProperties() {
348         Properties properties = new Properties();
349         properties.setProperty("windowTitle", windowTitle);
350         properties.setProperty("docTitle", docTitle);
351         properties.setProperty("footer", footer);
352         properties.setProperty("header", header);
353         properties.setProperty("author", Boolean.toString(displayAuthor));
354         properties.setProperty("overviewFile", overviewFile != null ? overviewFile.getAbsolutePath() : "");
355         try {
356             Scopes scopeVal = Scopes.valueOf(scope.toUpperCase());
357             if (scopeVal.equals(Scopes.PUBLIC)) {
358                 properties.setProperty("publicScope", "true");
359             } else if (scopeVal.equals(Scopes.PROTECTED)) {
360                 properties.setProperty("protectedScope", "true");
361             } else if (scopeVal.equals(Scopes.PACKAGE)) {
362                 properties.setProperty("packageScope", "true");
363             } else if (scopeVal.equals(Scopes.PRIVATE)) {
364                 properties.setProperty("privateScope", "true");
365             }
366         } catch (IllegalArgumentException e) {
367             getLog().warn("Scope (" + scope + ") was not recognized. Skipping argument.");
368         }
369 
370         return properties;
371     }

373     /**
374      * Sets up the GroovyDoc links.
375      *
376      * @return the GroovyDoc links
377      * @throws ClassNotFoundException    when a class needed for setting up GroovyDoc links cannot be found
378      * @throws InstantiationException    when a class needed for setting up GroovyDoc links cannot be instantiated
379      * @throws IllegalAccessException    when a method needed for setting up GroovyDoc links cannot be accessed
380      * @throws InvocationTargetException when a reflection invocation needed for setting up GroovyDoc links cannot be completed
381      */
382     @SuppressWarnings({"rawtypes", "unchecked"})
383     protected List<?> setupLinks() throws ClassNotFoundException, InvocationTargetException, IllegalAccessException, InstantiationException {
384         List linksList = new ArrayList();
385         if (links != null && links.size() > 0) {
386             Class<?> linkArgumentClass = null;
387             if (this.linkArgumentClass == null) {
388                 if (groovyAtLeast(GROOVY_1_6_0_RC2)) {
389                     linkArgumentClass = classWrangler.getClass("org.codehaus.groovy.tools.groovydoc.LinkArgument");
390                 } else if (groovyAtLeast(GROOVY_1_5_2)) {
391                     linkArgumentClass = classWrangler.getClass("org.codehaus.groovy.ant.Groovydoc$LinkArgument");
392                 }
393             } else {
394                 linkArgumentClass = classWrangler.getClass(this.linkArgumentClass);
395             }
396             if (linkArgumentClass != null) {
397                 Method setHref = findMethod(linkArgumentClass, "setHref", String.class);
398                 Method setPackages = findMethod(linkArgumentClass, "setPackages", String.class);
399                 for (Link link : links) {
400                     Object linkArgument = invokeConstructor(findConstructor(linkArgumentClass));
401                     invokeMethod(setHref, linkArgument, link.getHref());
402                     invokeMethod(setPackages, linkArgument, link.getPackages());
403                     linksList.add(linkArgument);
404                 }
405             } else {
406                 getLog().warn("Requested to use GroovyDoc links, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be 1.5.2 or newer). Ignoring links parameter.");
407             }
408         }
409 
410         return linksList;
411     }

413     /**
414      * Instantiates a new GroovyDocTool.
415      *
416      * @param groovyDocToolClass       the GroovyDocTool class
417      * @param resourceManagerClass     the ResourceManager lass
418      * @param docProperties            the documentation properties
419      * @param classpathResourceManager the ClasspathResourceManager for the GroovyDocTool
420      * @param sourceDirectories        the source directories for the GroovyDocTool
421      * @param groovyDocTemplateInfo    the GroovyDocTemplateInfo for the GroovyDocTool
422      * @param groovyDocLinks           the GroovyDoc links
423      * @return the GroovyDocTool to use in GroovyDoc generation
424      * @throws InstantiationException    when a class needed for setting up GroovyDoc tool cannot be instantiated
425      * @throws IllegalAccessException    when a method needed for setting up GroovyDoc tool cannot be accessed
426      * @throws InvocationTargetException when a reflection invocation needed for setting up GroovyDoc tool cannot be completed
427      */
428     protected Object createGroovyDocTool(final Class<?> groovyDocToolClass, final Class<?> resourceManagerClass, final Properties docProperties, final Object classpathResourceManager, final List<String> sourceDirectories, final GroovyDocTemplateInfo groovyDocTemplateInfo, final List<?> groovyDocLinks) throws InvocationTargetException, IllegalAccessException, InstantiationException {
429         Object groovyDocTool;
430         if (groovyAtLeast(GROOVY_1_6_0_RC2)) {
431             groovyDocTool = invokeConstructor(findConstructor(groovyDocToolClass, resourceManagerClass, String[].class, String[].class, String[].class, String[].class, List.class, Properties.class),
432                     classpathResourceManager,
433                     sourceDirectories.toArray(new String[0]),
434                     defaultDocTemplates == null ? groovyDocTemplateInfo.defaultDocTemplates() : defaultDocTemplates,
435                     defaultPackageTemplates == null ? groovyDocTemplateInfo.defaultPackageTemplates() : defaultPackageTemplates,
436                     defaultClassTemplates == null ? groovyDocTemplateInfo.defaultClassTemplates() : defaultClassTemplates,
437                     groovyDocLinks,
438                     docProperties
439             );
440         } else if (groovyAtLeast(GROOVY_1_5_2)) {
441             groovyDocTool = invokeConstructor(findConstructor(groovyDocToolClass, resourceManagerClass, String.class, String[].class, String[].class, String[].class, List.class),
442                     classpathResourceManager,
443                     sourceDirectories.get(0),
444                     defaultDocTemplates == null ? groovyDocTemplateInfo.defaultDocTemplates() : defaultDocTemplates,
445                     defaultPackageTemplates == null ? groovyDocTemplateInfo.defaultPackageTemplates() : defaultPackageTemplates,
446                     defaultClassTemplates == null ? groovyDocTemplateInfo.defaultClassTemplates() : defaultClassTemplates,
447                     groovyDocLinks
448             );
449             if (sourceDirectories.size() > 1) {
450                 getLog().warn("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support more than one GroovyDoc source directory (must be 1.6-RC-2 or newer). Only using first source directory (" + sourceDirectories.get(0) + ").");
451             }
452         } else {
453             groovyDocTool = invokeConstructor(findConstructor(groovyDocToolClass, resourceManagerClass, String.class, String[].class, String[].class, String[].class),
454                     classpathResourceManager,
455                     sourceDirectories.get(0),
456                     defaultDocTemplates == null ? groovyDocTemplateInfo.defaultDocTemplates() : defaultDocTemplates,
457                     defaultPackageTemplates == null ? groovyDocTemplateInfo.defaultPackageTemplates() : defaultPackageTemplates,
458                     defaultClassTemplates == null ? groovyDocTemplateInfo.defaultClassTemplates() : defaultClassTemplates
459             );
460             if (sourceDirectories.size() > 1) {
461                 getLog().warn("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support more than one GroovyDoc source directory (must be 1.6-RC-2 or newer). Only using first source directory (" + sourceDirectories.get(0) + ").");
462             }
463         }
464 
465         return groovyDocTool;
466     }

468     /**
469      * Gets the Groovy sources without the Java sources (since the Java sources don't have Javadoc).
470      *
471      * @param sourceDirectories the source directories to get the Groovy sources from
472      * @param fileSetManager    the FileSetmanager to use to get the included files
473      * @return the groovy sources
474      */
475     protected List<String> setupGroovyDocSources(final FileSet[] sourceDirectories, final FileSetManager fileSetManager) {
476         List<String> javaSources = new ArrayList<>();
477         List<String> groovySources = new ArrayList<>();
478         List<String> possibleGroovyStubs = new ArrayList<>();
479         for (FileSet sourceDirectory : sourceDirectories) {
480             String[] sources = fileSetManager.getIncludedFiles(sourceDirectory);
481             for (String source : sources) {
482                 if (source.endsWith(".java") && !javaSources.contains(source)) {
483                     javaSources.add(source);
484                 } else if (!groovySources.contains(source)) {
485                     groovySources.add(source);
486                     possibleGroovyStubs.add(source.replaceFirst("\\." + FileUtils.getFileExtension(source), ".java"));
487                 }
488             }
489         }
490         javaSources.removeAll(possibleGroovyStubs);
491         List<String> groovyDocSources = new ArrayList<>();
492         groovyDocSources.addAll(javaSources);
493         groovyDocSources.addAll(groovySources);
494 
495         return groovyDocSources;
496     }

498     /**
499      * Performs the GroovyDoc generation.
500      *
501      * @param outputDirectory    the directory to output the GroovyDoc to
502      * @param groovyDocToolClass the GroovyDocTool class
503      * @param outputToolClass    the OutputTool class
504      * @param fileOutputTool     the FileOutputTool to use for GroovyDoc generation
505      * @param groovyDocSources   the sources to
506      * @param groovyDocTool      the GroovyDocTool to use for GroovyDoc generation
507      * @throws IllegalAccessException    when a method needed for GroovyDoc generation cannot be accessed
508      * @throws InvocationTargetException when a reflection invocation needed for GroovyDoc generation cannot be completed
509      */
510     protected void generateGroovyDoc(final File outputDirectory, final Class<?> groovyDocToolClass, final Class<?> outputToolClass, final Object fileOutputTool, final List<String> groovyDocSources, final Object groovyDocTool) throws InvocationTargetException, IllegalAccessException {
511         getLog().debug("Adding sources to generate GroovyDoc for:");
512         if (getLog().isDebugEnabled()) {
513             for (String groovyDocSource : groovyDocSources) {
514                 getLog().debug("    " + groovyDocSource);
515             }
516         }
517         if (groovyAtLeast(GROOVY_1_6_0_RC2)) {
518             invokeMethod(findMethod(groovyDocToolClass, "add", List.class), groovyDocTool, groovyDocSources);
519         } else {
520             Method add = findMethod(groovyDocToolClass, "add", String.class);
521             for (String groovyDocSource : groovyDocSources) {
522                 invokeMethod(add, groovyDocTool, groovyDocSource);
523             }
524         }
525         invokeMethod(findMethod(groovyDocToolClass, "renderToOutput", outputToolClass, String.class), groovyDocTool, fileOutputTool, outputDirectory.getAbsolutePath());
526     }

528     /**
529      * Copies the stylesheet to the specified output directory.
530      *
531      * @param outputDirectory The output directory to copy the stylesheet to
532      */
533     protected void copyStylesheet(final File outputDirectory) {
534         getLog().info("Using stylesheet from " + stylesheetFile.getAbsolutePath() + ".");
535         try {
536             BufferedReader bufferedReader = null;
537             BufferedWriter bufferedWriter = null;
538             try {
539                 if (stylesheetEncoding != null) {
540                     bufferedReader = new BufferedReader(new InputStreamReader(new FileInputStream(stylesheetFile), stylesheetEncoding));
541                 } else {
542                     bufferedReader = new BufferedReader(new InputStreamReader(new FileInputStream(stylesheetFile)));
543                 }
544                 StringBuilder css = new StringBuilder();
545                 String line;
546                 while ((line = bufferedReader.readLine()) != null) {
547                     css.append(line).append("\n");
548                 }
549                 File outfile = new File(outputDirectory, "stylesheet.css");
550                 if (stylesheetEncoding != null) {
551                     bufferedWriter = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(outfile), stylesheetEncoding));
552                 } else {
553                     bufferedWriter = new BufferedWriter(new OutputStreamWriter(new FileOutputStream(outfile)));
554                 }
555                 bufferedWriter.write(css.toString());
556             } finally {
557                 FileUtils.closeQuietly(bufferedReader);
558                 FileUtils.closeQuietly(bufferedWriter);
559             }
560         } catch (IOException e) {
561             getLog().warn("Unable to copy specified stylesheet (" + stylesheetFile.getAbsolutePath() + ").");
562         }
563     }
