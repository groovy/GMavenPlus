145     /**
146      * Gets the set of included files from the specified source files or source directory (if sources are null).
147      *
148      * @param sources the sources to get the included files from
149      * @param sourceDirectory the source directory to fall back on if sources are null
150      *
151      * @return the included files from the specified sources
152      */
153     protected Set<File> getFiles(FileSet[] sources, String sourceDirectory) {
154         Set<File> files = new HashSet<File>();
155         FileSetManager fileSetManager = new FileSetManager(getLog());
156 
157         if (sources != null) {
158             for (FileSet fileSet : sources) {
159                 for (String include : Arrays.asList(fileSetManager.getIncludedFiles(fileSet))) {
160                     files.add(new File(project.getBasedir().getAbsolutePath() + File.separator + fileSet.getDirectory(), include));
161                 }
162             }
163         } else {
164             FileSet fileSet = new FileSet();
165             String directory = project.getBasedir().getAbsolutePath() + File.separator + "src" + File.separator + sourceDirectory + File.separator + "groovy";
166             fileSet.setDirectory(directory);
167             fileSet.setIncludes(Arrays.asList(DEFAULT_SOURCE_PATTERN));
168             for (String file : fileSetManager.getIncludedFiles(fileSet)) {
169                 files.add(new File(directory, file));
170             }
171         }
172 
173         return files;
174     }

214     /**
215      * Performs the stub generation on the specified source files.
216      *
217      * @param sources the sources to perform stub generation on
218      * @param outputDirectory the directory to write the stub files to
219      * @throws ClassNotFoundException
220      * @throws InvocationTargetException
221      * @throws IllegalAccessException
222      * @throws InstantiationException
223      */
224     protected synchronized void doStubGeneration(Set<File> sources, File outputDirectory) throws ClassNotFoundException, InvocationTargetException, IllegalAccessException, InstantiationException {
225         // get classes we need with reflection
226         Class<?> compilerConfigurationClass = Class.forName("org.codehaus.groovy.control.CompilerConfiguration");
227         Class<?> javaStubCompilationUnitClass = Class.forName("org.codehaus.groovy.tools.javac.JavaStubCompilationUnit");
228         Class<?> groovyClassLoaderClass = Class.forName("groovy.lang.GroovyClassLoader");
229         Class<?> phasesClass = Class.forName("org.codehaus.groovy.control.Phases");
230 
231         // set up compile options
232         Object compilerConfiguration = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(compilerConfigurationClass));
233         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setDebug", boolean.class), compilerConfiguration, debug);
234         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setVerbose", boolean.class), compilerConfiguration, verbose);
235         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setWarningLevel", int.class), compilerConfiguration, warningLevel);
236         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTolerance", int.class), compilerConfiguration, tolerance);
237         if (Version.parseFromString(getGroovyVersion()).compareTo(new Version(1, 5, 0)) >= 0) {
238             // if plugin only runs on 1.5, then can assume 1.5
239 //            ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetBytecode", String.class), compilerConfiguration, targetBytecode);
240             //
241             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetBytecode", String.class), compilerConfiguration, "1.5");
242         }
243         if (sourceEncoding != null) {
244             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setSourceEncoding", String.class), compilerConfiguration, sourceEncoding);
245         }
246         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetDirectory", String.class), compilerConfiguration, outputDirectory.getAbsolutePath());
247 
248         // setup groovyClassLoader
249         ClassLoader parent = ClassLoader.getSystemClassLoader();
250         Object groovyClassLoader = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyClassLoaderClass, ClassLoader.class, compilerConfigurationClass), parent, compilerConfiguration);
251         Object javaStubCompilationUnit = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(javaStubCompilationUnitClass, compilerConfigurationClass, groovyClassLoaderClass, File.class), compilerConfiguration, groovyClassLoader, outputDirectory);
252 
253         // add Groovy sources
254         /* I'm not sure why, but whenever I try to use the
255          * JavaStubCompilationUnit.addSources(File[]) to avoid the .groovy
256          * extension limitation, I get a "java.lang.IllegalArgumentException: wrong number of arguments".
257          * And doing a CompilerConfiguration.setScriptExtensions() doesn't make
258          * it recognize other extensions.  So for now, will use my DotGroovyFile hack.
259          */
260         getLog().debug("Adding Groovy to generate stubs for:");
261         for (File source : sources) {
262             getLog().debug("    " + source);
263             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(javaStubCompilationUnitClass, "addSource", File.class), javaStubCompilationUnit, new DotGroovyFile(source));
264         }
265 
266         // generate the stubs
267         Object convPhase = ReflectionUtils.getStaticField(ReflectionUtils.findField(phasesClass, "CONVERSION", int.class));
268         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(javaStubCompilationUnitClass, "compile", int.class), javaStubCompilationUnit, convPhase);
269 
270         // log generated stubs
271         getLog().debug("Generated " + sources.size() + "stubs.");
272     }

285     /**
286      * This is a fix for fix for http://jira.codehaus.org/browse/MGROOVY-187
287      * It modifies the dates of the created stubs to 1970, ensuring that the Java
288      * compiler will not come along and overwrite perfectly good compiled Groovy
289      * just because it has a newer source stub.  Basically, this prevents the
290      * stubs from causing a side effect with the Java compiler, but still allows
291      * the stubs to work with JavaDoc.  Ideally, the code for this should be
292      * added to the code that creates the stubs.
293      *
294      * @param stubs the files on which to reset the modified date
295      */
296     protected void resetStubModifiedDates(Set<File> stubs) {
297         for (File stub : stubs) {
298             boolean success = stub.setLastModified(0L);
299             if (!success) {
300                 getLog().warn("Unable to set modified time on stub " + stub.getAbsolutePath() + ".");
301             }
302         }
303     }
