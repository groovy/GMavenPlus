225     /**
226      * Performs the stub generation on the specified source files.
227      *
228      * @param stubSources     the sources to perform stub generation on
229      * @param classpath       The classpath to use for compilation
230      * @param outputDirectory the directory to write the stub files to
231      * @throws ClassNotFoundException    when a class needed for stub generation cannot be found
232      * @throws InstantiationException    when a class needed for stub generation cannot be instantiated
233      * @throws IllegalAccessException    when a method needed for stub generation cannot be accessed
234      * @throws InvocationTargetException when a reflection invocation needed for stub generation cannot be completed
235      * @throws MalformedURLException     when a classpath element provides a malformed URL
236      */
237     protected synchronized void doStubGeneration(final Set<File> stubSources, final List<?> classpath, final File outputDirectory) throws ClassNotFoundException, InvocationTargetException, IllegalAccessException, InstantiationException, MalformedURLException {
238         if (stubSources == null || stubSources.isEmpty()) {
239             getLog().info("No sources specified for stub generation. Skipping.");
240             return;
241         }
242 
243         setupClassWrangler(classpath, includeClasspath);
244 
245         logPluginClasspath();
246         classWrangler.logGroovyVersion(mojoExecution.getMojoDescriptor().getGoal());
247 
248         if (groovyVersionSupportsAction() && !skipBytecodeCheck) {
249             verifyGroovyVersionSupportsTargetBytecode();
250         } else {
251             getLog().error("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support stub generation. The minimum version of Groovy required is " + minGroovyVersion + ". Skipping stub generation.");
252             return;
253         }
254 
255         // get classes we need with reflection
256         Class<?> compilerConfigurationClass = classWrangler.getClass("org.codehaus.groovy.control.CompilerConfiguration");
257         Class<?> javaStubCompilationUnitClass = classWrangler.getClass("org.codehaus.groovy.tools.javac.JavaStubCompilationUnit");
258         Class<?> groovyClassLoaderClass = classWrangler.getClass("groovy.lang.GroovyClassLoader");
259 
260         // setup stub generation options
261         Object compilerConfiguration = setupCompilerConfiguration(outputDirectory, compilerConfigurationClass);
262         Object groovyClassLoader = invokeConstructor(findConstructor(groovyClassLoaderClass, ClassLoader.class, compilerConfigurationClass), classWrangler.getClassLoader(), compilerConfiguration);
263         Object javaStubCompilationUnit = invokeConstructor(findConstructor(javaStubCompilationUnitClass, compilerConfigurationClass, groovyClassLoaderClass, File.class), compilerConfiguration, groovyClassLoader, outputDirectory);
264 
265         // add Groovy sources
266         addGroovySources(stubSources, compilerConfigurationClass, javaStubCompilationUnitClass, compilerConfiguration, javaStubCompilationUnit);
267 
268         // generate the stubs
269         invokeMethod(findMethod(javaStubCompilationUnitClass, "compile"), javaStubCompilationUnit);
270     }

272     /**
273      * Sets up the CompilerConfiguration to use for stub generation.
274      *
275      * @param outputDirectory            the directory to write the stub files to
276      * @param compilerConfigurationClass the CompilerConfiguration class
277      * @return the CompilerConfiguration to use for stub generation
278      * @throws InstantiationException    when a class needed for stub generation cannot be instantiated
279      * @throws IllegalAccessException    when a method needed for stub generation cannot be accessed
280      * @throws InvocationTargetException when a reflection invocation needed for stub generation cannot be completed
281      */
282     protected Object setupCompilerConfiguration(final File outputDirectory, final Class<?> compilerConfigurationClass) throws InvocationTargetException, IllegalAccessException, InstantiationException {
283         Object compilerConfiguration = invokeConstructor(findConstructor(compilerConfigurationClass));
284         invokeMethod(findMethod(compilerConfigurationClass, "setDebug", boolean.class), compilerConfiguration, debug);
285         invokeMethod(findMethod(compilerConfigurationClass, "setVerbose", boolean.class), compilerConfiguration, verbose);
286         invokeMethod(findMethod(compilerConfigurationClass, "setWarningLevel", int.class), compilerConfiguration, warningLevel);
287         invokeMethod(findMethod(compilerConfigurationClass, "setTolerance", int.class), compilerConfiguration, tolerance);
288         invokeMethod(findMethod(compilerConfigurationClass, "setTargetBytecode", String.class), compilerConfiguration, targetBytecode);
289         if (sourceEncoding != null) {
290             invokeMethod(findMethod(compilerConfigurationClass, "setSourceEncoding", String.class), compilerConfiguration, sourceEncoding);
291         }
292         Map<String, Object> options = new HashMap<>();
293         options.put("stubDir", outputDirectory);
294         options.put("keepStubs", Boolean.TRUE);
295         invokeMethod(findMethod(compilerConfigurationClass, "setJointCompilationOptions", Map.class), compilerConfiguration, options);
296 
297         return compilerConfiguration;
298     }

300     /**
301      * Adds the Groovy sources to the CompilationUnit.
302      *
303      * @param stubSources                  the sources to perform stub generation on
304      * @param compilerConfigurationClass   the CompilerConfiguration class
305      * @param javaStubCompilationUnitClass the JavaStubCompilationUnit class
306      * @param compilerConfiguration        the CompilerConfiguration to use for stub generation
307      * @param javaStubCompilationUnit      the JavaStubCompilationUnit to use for stub generation
308      * @throws IllegalAccessException    when a method needed for stub generation cannot be accessed
309      * @throws InvocationTargetException when a reflection invocation needed for stub generation cannot be completed
310      */
311     protected void addGroovySources(final Set<File> stubSources, final Class<?> compilerConfigurationClass, final Class<?> javaStubCompilationUnitClass, final Object compilerConfiguration, final Object javaStubCompilationUnit) throws InvocationTargetException, IllegalAccessException {
312         Set<String> scriptExtensions = new HashSet<>();
313         for (File stubSource : stubSources) {
314             scriptExtensions.add(FileUtils.getFileExtension(stubSource));
315         }
316         if (getLog().isDebugEnabled()) {
317             getLog().debug("Detected Groovy file extensions: " + scriptExtensions + ".");
318         }
319         if (supportsSettingExtensions()) {
320             invokeMethod(findMethod(compilerConfigurationClass, "setScriptExtensions", Set.class), compilerConfiguration, scriptExtensions);
321         }
322         getLog().debug("Adding Groovy to generate stubs for:");
323         Method addSource = findMethod(javaStubCompilationUnitClass, "addSource", File.class);
324         for (File stubSource : stubSources) {
325             if (getLog().isDebugEnabled()) {
326                 getLog().debug("    " + stubSource);
327             }
328             if (supportsSettingExtensions()) {
329                 invokeMethod(addSource, javaStubCompilationUnit, stubSource);
330             } else {
331                 DotGroovyFileDotGroovyFile.html#DotGroovyFile">DotGroovyFile dotGroovyFile = new DotGroovyFile(stubSource).setScriptExtensions(scriptExtensions);
332                 invokeMethod(addSource, javaStubCompilationUnit, dotGroovyFile);
333             }
334         }
335     }

358     /**
359      * This is a fix for http://jira.codehaus.org/browse/MGROOVY-187
360      * It modifies the dates of the created stubs to 1/1/1970, ensuring that the Java compiler will not overwrite perfectly
361      * good compiled Groovy just because it has a newer source stub. Basically, this prevents the stubs from causing a
362      * side-effect with the Java compiler, but still allows stubs to work with JavaDoc.
363      *
364      * @param stubs the files on which to reset the modified date
365      */
366     protected void resetStubModifiedDates(final Set<File> stubs) {
367         for (File stub : stubs) {
368             boolean success = stub.setLastModified(0L);
369             if (!success) {
370                 if (getLog().isWarnEnabled()) {
371                     getLog().warn("Unable to set modified time on stub " + stub.getAbsolutePath() + ".");
372                 }
373             }
374         }
375     }

377     /**
378      * Throws an exception if targetBytecode is not supported with this version of Groovy. That is, when Groovy added
379      * the option to org.codehaus.groovy.control.CompilerConfiguration and used it in
380      * org.codehaus.groovy.classgen.asm.WriterController.
381      */
382     protected void verifyGroovyVersionSupportsTargetBytecode() {
383         if ("16".equals(targetBytecode)) {
384             if (groovyOlderThan(GROOVY_3_0_6)) {
385                 throw new IllegalArgumentException("Target bytecode 16 requires Groovy " + GROOVY_3_0_6 + " or newer.");
386             }
387         } else if ("15".equals(targetBytecode)) {
388             if (groovyOlderThan(GROOVY_3_0_3)) {
389                 throw new IllegalArgumentException("Target bytecode 15 requires Groovy " + GROOVY_3_0_3 + " or newer.");
390             }
391         } else if ("14".equals(targetBytecode)) {
392             if (groovyOlderThan(GROOVY_3_0_0_BETA2)) {
393                 throw new IllegalArgumentException("Target bytecode 14 requires Groovy " + GROOVY_3_0_0_BETA2 + " or newer.");
394             }
395         } else if ("13".equals(targetBytecode)) {
396             if (groovyOlderThan(GROOVY_2_5_7) || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_BETA1))) {
397                 throw new IllegalArgumentException("Target bytecode 13 requires Groovy " + GROOVY_2_5_7 + "/" + GROOVY_3_0_0_BETA1 + " or newer. No 2.6 version is supported.");
398             }
399         } else if ("12".equals(targetBytecode) || "11".equals(targetBytecode) || "10".equals(targetBytecode)) {
400             if (groovyOlderThan(GROOVY_2_5_3) || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_ALPHA4))) {
401                 throw new IllegalArgumentException("Target bytecode 10, 11, or 12 requires Groovy " + GROOVY_2_5_3 + "/" + GROOVY_3_0_0_ALPHA4 + " or newer. No 2.6 version is supported.");
402             }
403         } else if ("9".equals(targetBytecode)) {
404             if (!isGroovyIndy() && (groovyOlderThan(GROOVY_2_5_3)
405                     || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_2_6_0_ALPHA4))
406                     || (groovyAtLeast(GROOVY_3_0_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_ALPHA2)))) {
407                 throw new IllegalArgumentException("Target bytecode 9 requires Groovy " + GROOVY_2_5_3 + "/" + GROOVY_2_6_0_ALPHA4 + "/" + GROOVY_3_0_0_ALPHA2 + " or newer.");
408             } else if (isGroovyIndy() && (groovyOlderThan(GROOVY_2_5_3) || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_ALPHA4)))) {
409                 throw new IllegalArgumentException("Target bytecode 9 with invokedynamic requires Groovy " + GROOVY_2_5_3 + "/" + GROOVY_3_0_0_ALPHA4 + " or newer. No 2.6 version is supported.");
410             }
411         } else if ("1.8".equals(targetBytecode)) {
412             if (groovyOlderThan(GROOVY_2_3_3)) {
413                 throw new IllegalArgumentException("Target bytecode 1.8 requires Groovy " + GROOVY_2_3_3 + " or newer.");
414             }
415         } else if ("1.7".equals(targetBytecode) || "1.6".equals(targetBytecode)) {
416             if (groovyOlderThan(GROOVY_2_1_3)) {
417                 throw new IllegalArgumentException("Target bytecode 1.6 and 1.7 require Groovy " + GROOVY_2_1_3 + " or newer.");
418             }
419         } else if (!"1.5".equals(targetBytecode) && !"1.4".equals(targetBytecode)) {
420             throw new IllegalArgumentException("Unrecognized target bytecode: '" + targetBytecode + "'.");
421         }
422     }
