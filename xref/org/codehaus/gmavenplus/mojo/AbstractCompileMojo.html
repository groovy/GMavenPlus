47       * Groovy 2.1.0 beta-1 version.
48       */

151     /**
152      * Performs compilation of compile mojos.
153      *
154      * @param sources the sources to compile
155      * @param classpath the classpath to use for compilation
156      * @param compileOutputDirectory the directory to write the compiled class files to
157      * @throws ClassNotFoundException when a class needed for compilation cannot be found
158      * @throws InstantiationException when a class needed for compilation cannot be instantiated
159      * @throws IllegalAccessException when a method needed for compilation cannot be accessed
160      * @throws InvocationTargetException when a reflection invocation needed for compilation cannot be completed
161      * @throws MalformedURLException when a classpath element provides a malformed URL
162      */
163     @SuppressWarnings("unchecked")
164     protected synchronized void doCompile(final Set<File> sources, final List classpath, final File compileOutputDirectory)
165             throws ClassNotFoundException, InstantiationException, IllegalAccessException, InvocationTargetException, MalformedURLException {
166         classWrangler = new ClassWrangler(classpath, getLog());
167 
168         if (sources == null || sources.isEmpty()) {
169             getLog().info("No sources specified for compilation.  Skipping.");
170             return;
171         }
172         if (groovyVersionSupportsAction()) {
173             classWrangler.logGroovyVersion(mojoExecution.getMojoDescriptor().getGoal());
174             logPluginClasspath();
175             if (getLog().isDebugEnabled()) {
176                 try {
177                     getLog().debug("Project compile classpath:\n" + project.getCompileClasspathElements());
178                 } catch (DependencyResolutionRequiredException e) {
179                     getLog().warn("Unable to log project compile classpath", e);
180                 }
181             }
182         } else {
183             getLog().error("Your Groovy version (" + classWrangler.getGroovyVersion() + ") doesn't support compilation.  The minimum version of Groovy required is " + minGroovyVersion + ".  Skipping compiling.");
184             return;
185         }
186 
187         // get classes we need with reflection
188         Class compilerConfigurationClass = classWrangler.getClass("org.codehaus.groovy.control.CompilerConfiguration");
189         Class compilationUnitClass = classWrangler.getClass("org.codehaus.groovy.control.CompilationUnit");
190         Class groovyClassLoaderClass = classWrangler.getClass("groovy.lang.GroovyClassLoader");
191 
192         // setup compile options
193         Object compilerConfiguration = setupCompilerConfiguration(compileOutputDirectory, compilerConfigurationClass);
194         Object groovyClassLoader = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyClassLoaderClass, ClassLoader.class, compilerConfigurationClass), classWrangler.getClassLoader(), compilerConfiguration);
195         Object transformLoader = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyClassLoaderClass, ClassLoader.class), classWrangler.getClassLoader());
196 
197         // add Groovy sources
198         Object compilationUnit = setupCompilationUnit(sources, compilerConfigurationClass, compilationUnitClass, groovyClassLoaderClass, compilerConfiguration, groovyClassLoader, transformLoader);
199 
200         // compile the classes
201         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilationUnitClass, "compile"), compilationUnit);
202 
203         // log compiled classes
204         List classes = (List) ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilationUnitClass, "getClasses"), compilationUnit);
205         getLog().info("Compiled " + classes.size() + " file" + (classes.size() > 1 || classes.size() == 0 ? "s" : "") + ".");
206     }

208     /**
209      * Sets up the CompilationUnit to use for compilation.
210      *
211      * @param sources the sources to compile
212      * @param compilerConfigurationClass the CompilerConfiguration class
213      * @param compilationUnitClass the CompilationUnit class
214      * @param groovyClassLoaderClass the GroovyClassLoader class
215      * @param compilerConfiguration the CompilerConfiguration
216      * @param groovyClassLoader the GroovyClassLoader
217      * @param transformLoader the GroovyClassLoader to use for transformation
218      * @return the CompilationUnit
219      * @throws InstantiationException when a class needed for setting up compilation unit cannot be instantiated
220      * @throws IllegalAccessException when a method needed for setting up compilation unit cannot be accessed
221      * @throws InvocationTargetException when a reflection invocation needed for setting up compilation unit cannot be completed
222      */
223     protected Object setupCompilationUnit(final Set<File> sources, final Class compilerConfigurationClass, final Class compilationUnitClass, final Class groovyClassLoaderClass, final Object compilerConfiguration, final Object groovyClassLoader, final Object transformLoader) throws InvocationTargetException, IllegalAccessException, InstantiationException {
224         Object compilationUnit;
225         if (classWrangler.getGroovyVersion().compareTo(GROOVY_1_6_0) >= 0) {
226             compilationUnit = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(compilationUnitClass, compilerConfigurationClass, CodeSource.class, groovyClassLoaderClass, groovyClassLoaderClass), compilerConfiguration, null, groovyClassLoader, transformLoader);
227         } else {
228             compilationUnit = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(compilationUnitClass, compilerConfigurationClass, CodeSource.class, groovyClassLoaderClass), compilerConfiguration, null, groovyClassLoader);
229         }
230         getLog().debug("Adding Groovy to compile:");
231         for (File source : sources) {
232             if (getLog().isDebugEnabled()) {
233                 getLog().debug("    " + source);
234             }
235             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilationUnitClass, "addSource", File.class), compilationUnit, source);
236         }
237 
238         return compilationUnit;
239     }

241     /**
242      * Sets up the CompilationConfiguration to use for compilation.
243      *
244      * @param compileOutputDirectory the directory to write the compiled classes to
245      * @param compilerConfigurationClass the CompilerConfiguration class
246      * @return the CompilerConfiguration
247      * @throws ClassNotFoundException when a class needed for setting up CompilerConfiguration cannot be found
248      * @throws InstantiationException when a class needed for setting up CompilerConfiguration cannot be instantiated
249      * @throws IllegalAccessException when a method needed for setting up CompilerConfiguration cannot be accessed
250      * @throws InvocationTargetException when a reflection invocation needed for setting up CompilerConfiguration cannot be completed
251      */
252     @SuppressWarnings("unchecked")
253     protected Object setupCompilerConfiguration(final File compileOutputDirectory, final Class compilerConfigurationClass) throws InvocationTargetException, IllegalAccessException, InstantiationException, ClassNotFoundException {
254         Object compilerConfiguration = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(compilerConfigurationClass));
255         if (configScript != null) {
256             if (classWrangler.getGroovyVersion().compareTo(GROOVY_2_1_0_BETA1) >= 0) {
257                 Class bindingClass = classWrangler.getClass("groovy.lang.Binding");
258                 Class importCustomizerClass = classWrangler.getClass("org.codehaus.groovy.control.customizers.ImportCustomizer");
259                 Class groovyShellClass = classWrangler.getClass("groovy.lang.GroovyShell");
260 
261                 Object binding = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(bindingClass));
262                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(bindingClass, "setVariable", String.class, Object.class), binding, "configuration", compilerConfiguration);
263                 Object shellCompilerConfiguration = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(compilerConfigurationClass));
264                 Object importCustomizer = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(importCustomizerClass));
265                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(importCustomizerClass, "addStaticStar", String.class), importCustomizer, "org.codehaus.groovy.control.customizers.builder.CompilerCustomizationBuilder");
266                 List compilationCustomizers = (List) ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "getCompilationCustomizers"), shellCompilerConfiguration);
267                 compilationCustomizers.add(importCustomizer);
268                 Object shell = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyShellClass, bindingClass, compilerConfigurationClass), binding, shellCompilerConfiguration);
269                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyShellClass, "evaluate", File.class), shell , configScript);
270             } else {
271                 getLog().warn("Requested to use configScript, but your Groovy version doesn't support it (must be 2.1.0-beta-1 or newer).  Ignoring configScript parameter.");
272             }
273         }
274         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setDebug", boolean.class), compilerConfiguration, debug);
275         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setVerbose", boolean.class), compilerConfiguration, verbose);
276         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setWarningLevel", int.class), compilerConfiguration, warningLevel);
277         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTolerance", int.class), compilerConfiguration, tolerance);
278         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetBytecode", String.class), compilerConfiguration, targetBytecode);
279         if (sourceEncoding != null) {
280             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setSourceEncoding", String.class), compilerConfiguration, sourceEncoding);
281         }
282         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetDirectory", String.class), compilerConfiguration, compileOutputDirectory.getAbsolutePath());
283         if (invokeDynamic) {
284             if (classWrangler.getGroovyVersion().compareTo(GROOVY_2_1_0_BETA3) >= 0) {
285                 if (classWrangler.isGroovyIndy()) {
286                     if (isJavaSupportIndy()) {
287                         Map<String, Boolean> optimizationOptions = (Map<String, Boolean>) ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "getOptimizationOptions"), compilerConfiguration);
288                         optimizationOptions.put("indy", true);
289                         optimizationOptions.put("int", false);
290                     } else {
291                         getLog().warn("Tried to use to use an indy version of Groovy, but your Java version (" + getJavaVersionString() + ") doesn't support it.  Ignoring invokeDynamic parameter.");
292                     }
293                 } else {
294                     getLog().warn("Requested to use InvokeDynamic, but the version of Groovy on the project classpath doesn't support it (must use have indy classifier).  Ignoring invokeDynamic parameter.");
295                 }
296             } else {
297                 getLog().warn("Requested to use invokeDynamic, but your Groovy version doesn't support it (must be 2.0.0-beta-3 or newer).  Ignoring invokeDynamic parameter.");
298             }
299         }
300 
301         return compilerConfiguration;
302     }
