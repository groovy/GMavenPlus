242     /**
243      * Performs compilation of compile mojos.
244      *
245      * @param sources                the sources to compile
246      * @param classpath              the classpath to use for compilation
247      * @param compileOutputDirectory the directory to write the compiled class files to
248      * @throws ClassNotFoundException    when a class needed for compilation cannot be found
249      * @throws InstantiationException    when a class needed for compilation cannot be instantiated
250      * @throws IllegalAccessException    when a method needed for compilation cannot be accessed
251      * @throws InvocationTargetException when a reflection invocation needed for compilation cannot be completed
252      * @throws MalformedURLException     when a classpath element provides a malformed URL
253      */
254     @SuppressWarnings({"rawtypes"})
255     protected synchronized void doCompile(final Set<File> sources, final List classpath, final File compileOutputDirectory)
256             throws ClassNotFoundException, InstantiationException, IllegalAccessException, InvocationTargetException, MalformedURLException {
257         if (sources == null || sources.isEmpty()) {
258             getLog().info("No sources specified for compilation. Skipping.");
259             return;
260         }
261 
262         setupClassWrangler(classpath, includeClasspath);
263 
264         logPluginClasspath();
265         classWrangler.logGroovyVersion(mojoExecution.getMojoDescriptor().getGoal());
266 
267         if (groovyVersionSupportsAction() && !skipBytecodeCheck) {
268             verifyGroovyVersionSupportsTargetBytecode();
269         } else {
270             getLog().error("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support compilation. The minimum version of Groovy required is " + minGroovyVersion + ". Skipping compiling.");
271             return;
272         }
273 
274         // get classes we need with reflection
275         Class<?> compilerConfigurationClass = classWrangler.getClass("org.codehaus.groovy.control.CompilerConfiguration");
276         Class<?> compilationUnitClass = classWrangler.getClass("org.codehaus.groovy.control.CompilationUnit");
277         Class<?> groovyClassLoaderClass = classWrangler.getClass("groovy.lang.GroovyClassLoader");
278 
279         // setup compile options
280         Object compilerConfiguration = setupCompilerConfiguration(compileOutputDirectory, compilerConfigurationClass);
281         Object groovyClassLoader = invokeConstructor(findConstructor(groovyClassLoaderClass, ClassLoader.class, compilerConfigurationClass), classWrangler.getClassLoader(), compilerConfiguration);
282         Object transformLoader = invokeConstructor(findConstructor(groovyClassLoaderClass, ClassLoader.class), classWrangler.getClassLoader());
283 
284         // add Groovy sources
285         Object compilationUnit = setupCompilationUnit(sources, compilerConfigurationClass, compilationUnitClass, groovyClassLoaderClass, compilerConfiguration, groovyClassLoader, transformLoader);
286 
287         // compile the classes
288         invokeMethod(findMethod(compilationUnitClass, "compile"), compilationUnit);
289 
290         // log compiled classes
291         List classes = (List) invokeMethod(findMethod(compilationUnitClass, "getClasses"), compilationUnit);
292         getLog().info("Compiled " + classes.size() + " file" + (classes.size() != 1 ? "s" : "") + ".");
293     }

295     /**
296      * Sets up the CompilationUnit to use for compilation.
297      *
298      * @param sources                    the sources to compile
299      * @param compilerConfigurationClass the CompilerConfiguration class
300      * @param compilationUnitClass       the CompilationUnit class
301      * @param groovyClassLoaderClass     the GroovyClassLoader class
302      * @param compilerConfiguration      the CompilerConfiguration
303      * @param groovyClassLoader          the GroovyClassLoader
304      * @param transformLoader            the GroovyClassLoader to use for transformation
305      * @return the CompilationUnit
306      * @throws InstantiationException    when a class needed for setting up compilation unit cannot be instantiated
307      * @throws IllegalAccessException    when a method needed for setting up compilation unit cannot be accessed
308      * @throws InvocationTargetException when a reflection invocation needed for setting up compilation unit cannot be completed
309      */
310     protected Object setupCompilationUnit(final Set<File> sources, final Class<?> compilerConfigurationClass, final Class<?> compilationUnitClass, final Class<?> groovyClassLoaderClass, final Object compilerConfiguration, final Object groovyClassLoader, final Object transformLoader) throws InvocationTargetException, IllegalAccessException, InstantiationException {
311         Object compilationUnit;
312         if (groovyAtLeast(GROOVY_1_6_0)) {
313             compilationUnit = invokeConstructor(findConstructor(compilationUnitClass, compilerConfigurationClass, CodeSource.class, groovyClassLoaderClass, groovyClassLoaderClass), compilerConfiguration, null, groovyClassLoader, transformLoader);
314         } else {
315             compilationUnit = invokeConstructor(findConstructor(compilationUnitClass, compilerConfigurationClass, CodeSource.class, groovyClassLoaderClass), compilerConfiguration, null, groovyClassLoader);
316         }
317         getLog().debug("Adding Groovy to compile:");
318         Method addSourceMethod = findMethod(compilationUnitClass, "addSource", File.class);
319         for (File source : sources) {
320             getLog().debug("    " + source);
321             invokeMethod(addSourceMethod, compilationUnit, source);
322         }
323 
324         return compilationUnit;
325     }

327     /**
328      * Sets up the CompilationConfiguration to use for compilation.
329      *
330      * @param compileOutputDirectory     the directory to write the compiled classes to
331      * @param compilerConfigurationClass the CompilerConfiguration class
332      * @return the CompilerConfiguration
333      * @throws ClassNotFoundException    when a class needed for setting up CompilerConfiguration cannot be found
334      * @throws InstantiationException    when a class needed for setting up CompilerConfiguration cannot be instantiated
335      * @throws IllegalAccessException    when a method needed for setting up CompilerConfiguration cannot be accessed
336      * @throws InvocationTargetException when a reflection invocation needed for setting up CompilerConfiguration cannot be completed
337      */
338     @SuppressWarnings({"rawtypes", "unchecked"})
339     protected Object setupCompilerConfiguration(final File compileOutputDirectory, final Class<?> compilerConfigurationClass) throws InvocationTargetException, IllegalAccessException, InstantiationException, ClassNotFoundException {
340         Object compilerConfiguration = invokeConstructor(findConstructor(compilerConfigurationClass));
341         if (configScript != null) {
342             if (!configScript.exists()) {
343                 getLog().warn("Configuration script file (" + configScript.getAbsolutePath() + ") doesn't exist. Ignoring configScript parameter.");
344             } else if (groovyOlderThan(GROOVY_2_1_0_BETA1)) {
345                 getLog().warn("Requested to use configScript, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_2_1_0_BETA1 + " or newer). Ignoring configScript parameter.");
346             } else {
347                 Class<?> bindingClass = classWrangler.getClass("groovy.lang.Binding");
348                 Class<?> importCustomizerClass = classWrangler.getClass("org.codehaus.groovy.control.customizers.ImportCustomizer");
349                 Class<?> groovyShellClass = classWrangler.getClass("groovy.lang.GroovyShell");
350 
351                 Object binding = invokeConstructor(findConstructor(bindingClass));
352                 invokeMethod(findMethod(bindingClass, "setVariable", String.class, Object.class), binding, "configuration", compilerConfiguration);
353                 Object shellCompilerConfiguration = invokeConstructor(findConstructor(compilerConfigurationClass));
354                 Object importCustomizer = invokeConstructor(findConstructor(importCustomizerClass));
355                 invokeMethod(findMethod(importCustomizerClass, "addStaticStar", String.class), importCustomizer, "org.codehaus.groovy.control.customizers.builder.CompilerCustomizationBuilder");
356                 List compilationCustomizers = (List) invokeMethod(findMethod(compilerConfigurationClass, "getCompilationCustomizers"), shellCompilerConfiguration);
357                 compilationCustomizers.add(importCustomizer);
358                 Object shell = invokeConstructor(findConstructor(groovyShellClass, bindingClass, compilerConfigurationClass), binding, shellCompilerConfiguration);
359                 getLog().debug("Using configuration script " + configScript + " for compilation.");
360                 invokeMethod(findMethod(groovyShellClass, "evaluate", File.class), shell, configScript);
361             }
362         }
363         invokeMethod(findMethod(compilerConfigurationClass, "setDebug", boolean.class), compilerConfiguration, debug);
364         invokeMethod(findMethod(compilerConfigurationClass, "setVerbose", boolean.class), compilerConfiguration, verbose);
365         invokeMethod(findMethod(compilerConfigurationClass, "setWarningLevel", int.class), compilerConfiguration, warningLevel);
366         invokeMethod(findMethod(compilerConfigurationClass, "setTolerance", int.class), compilerConfiguration, tolerance);
367         invokeMethod(findMethod(compilerConfigurationClass, "setTargetBytecode", String.class), compilerConfiguration, targetBytecode);
368         if (previewFeatures) {
369             if (isJavaSupportPreviewFeatures()) {
370                 if (groovyOlderThan(GROOVY_2_5_7) || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_BETA1))) {
371                     getLog().warn("Requested to use preview features, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_2_5_7 + "/" + GROOVY_3_0_0_BETA1 + " or newer. No 2.6 version is supported. Ignoring previewFeatures parameter.");
372                 } else {
373                     invokeMethod(findMethod(compilerConfigurationClass, "setPreviewFeatures", boolean.class), compilerConfiguration, previewFeatures);
374                 }
375             } else {
376                 getLog().warn("Requested to use to use preview features, but your Java version (" + getJavaVersionString() + ") doesn't support it. Ignoring previewFeatures parameter.");
377             }
378         }
379         if (sourceEncoding != null) {
380             invokeMethod(findMethod(compilerConfigurationClass, "setSourceEncoding", String.class), compilerConfiguration, sourceEncoding);
381         }
382         invokeMethod(findMethod(compilerConfigurationClass, "setTargetDirectory", String.class), compilerConfiguration, compileOutputDirectory.getAbsolutePath());
383         if (invokeDynamic) {
384             if (groovyAtLeast(GROOVY_2_0_0_BETA3)) {
385                 if (classWrangler.isGroovyIndy()) {
386                     if (isJavaSupportIndy()) {
387                         Map<String, Boolean> optimizationOptions = (Map<String, Boolean>) invokeMethod(findMethod(compilerConfigurationClass, "getOptimizationOptions"), compilerConfiguration);
388                         optimizationOptions.put("indy", true);
389                         optimizationOptions.put("int", false);
390                     } else {
391                         getLog().warn("Requested to use to use invokedynamic, but your Java version (" + getJavaVersionString() + ") doesn't support it. Ignoring invokeDynamic parameter.");
392                     }
393                 } else {
394                     getLog().warn("Requested to use invokedynamic, but your Groovy version doesn't support it (must use have indy classifier). Ignoring invokeDynamic parameter.");
395                 }
396             } else {
397                 getLog().warn("Requested to use invokeDynamic, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_2_0_0_BETA3 + " or newer). Ignoring invokeDynamic parameter.");
398             }
399         }
400         if (parameters) {
401             if (groovyAtLeast(GROOVY_2_5_0_ALPHA1)) {
402                 if (isJavaSupportParameters()) {
403                     invokeMethod(findMethod(compilerConfigurationClass, "setParameters", boolean.class), compilerConfiguration, parameters);
404                 } else {
405                     getLog().warn("Requested to use to use parameters, but your Java version (" + getJavaVersionString() + ") doesn't support it. Ignoring parameters parameter.");
406                 }
407             } else {
408                 getLog().warn("Requested to use parameters, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_2_5_0_ALPHA1 + " or newer). Ignoring parameters parameter.");
409             }
410         }
411 
412         return compilerConfiguration;
413     }

415     /**
416      * Throws an exception if targetBytecode is not supported with this version of Groovy. That is, when Groovy added
417      * the option to org.codehaus.groovy.control.CompilerConfiguration and used it in
418      * org.codehaus.groovy.classgen.asm.WriterController.
419      */
420     protected void verifyGroovyVersionSupportsTargetBytecode() {
421         if ("15".equals(targetBytecode)) {
422             if (groovyOlderThan(GROOVY_3_0_3)) {
423                 throw new IllegalArgumentException("Target bytecode 15 requires Groovy " + GROOVY_3_0_3 + " or newer.");
424             }
425         } else if ("14".equals(targetBytecode)) {
426             if (groovyOlderThan(GROOVY_3_0_0_BETA2)) {
427                 throw new IllegalArgumentException("Target bytecode 14 requires Groovy " + GROOVY_3_0_0_BETA2 + " or newer.");
428             }
429         } else if ("13".equals(targetBytecode)) {
430             if (groovyOlderThan(GROOVY_2_5_7) || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_BETA1))) {
431                 throw new IllegalArgumentException("Target bytecode 13 requires Groovy " + GROOVY_2_5_7 + "/" + GROOVY_3_0_0_BETA1 + " or newer. No 2.6 version is supported.");
432             }
433         } else if ("12".equals(targetBytecode) || "11".equals(targetBytecode) || "10".equals(targetBytecode)) {
434             if (groovyOlderThan(GROOVY_2_5_3) || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_ALPHA4))) {
435                 throw new IllegalArgumentException("Target bytecode 10, 11, or 12 requires Groovy " + GROOVY_2_5_3 + "/" + GROOVY_3_0_0_ALPHA4 + " or newer. No 2.6 version is supported.");
436             }
437         } else if ("9".equals(targetBytecode)) {
438             if (!isGroovyIndy() && (groovyOlderThan(GROOVY_2_5_3)
439                     || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_2_6_0_ALPHA4))
440                     || (groovyAtLeast(GROOVY_3_0_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_ALPHA2)))) {
441                 throw new IllegalArgumentException("Target bytecode 9 requires Groovy " + GROOVY_2_5_3 + "/" + GROOVY_2_6_0_ALPHA4 + "/" + GROOVY_3_0_0_ALPHA2 + " or newer.");
442             } else if (isGroovyIndy() && (groovyOlderThan(GROOVY_2_5_3) || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_ALPHA4)))) {
443                 throw new IllegalArgumentException("Target bytecode 9 with invokedynamic requires Groovy " + GROOVY_2_5_3 + "/" + GROOVY_3_0_0_ALPHA4 + " or newer. No 2.6 version is supported.");
444             }
445         } else if ("1.8".equals(targetBytecode)) {
446             if (groovyOlderThan(GROOVY_2_3_3)) {
447                 throw new IllegalArgumentException("Target bytecode 1.8 requires Groovy " + GROOVY_2_3_3 + " or newer.");
448             }
449         } else if ("1.7".equals(targetBytecode) || "1.6".equals(targetBytecode)) {
450             if (groovyOlderThan(GROOVY_2_1_3)) {
451                 throw new IllegalArgumentException("Target bytecode 1.6 and 1.7 require Groovy " + GROOVY_2_1_3 + " or newer.");
452             }
453         } else if (!"1.5".equals(targetBytecode) && !"1.4".equals(targetBytecode)) {
454             throw new IllegalArgumentException("Unrecognized target bytecode: '" + targetBytecode + "'.");
455         }
456     }
