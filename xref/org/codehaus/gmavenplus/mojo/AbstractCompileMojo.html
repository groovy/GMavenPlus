17  package org.codehaus.gmavenplus.mojo;
18  
19  import org.apache.maven.plugin.MojoExecutionException;
20  import org.codehaus.gmavenplus.model.Version;
21  import org.codehaus.gmavenplus.util.ReflectionUtils;
22  import java.io.File;
23  import java.lang.reflect.InvocationTargetException;
24  import java.net.MalformedURLException;
25  import java.net.URL;
26  import java.security.CodeSource;
27  import java.util.*;
28  
29  
30  /**
31   * The base compile mojo, which all compile mojos extend.
32   *
33   * @author Keegan Witt
34   */
35  public abstract class AbstractCompileMojo extends AbstractGroovySourcesMojo {
36  
37      /**
38       * The minimum version of Groovy that this mojo supports.
39       */
40      protected static final Version MIN_GROOVY_VERSION = new Version(1, 5, 0);
41  
42      /**
43       * The location for the compiled classes.
44       *
45       * @parameter default-value="${project.build.outputDirectory}"
46       */
47      protected File outputDirectory;
48  
49      /**
50       * The location for the compiled test classes.
51       *
52       * @parameter default-value="${project.build.testOutputDirectory}"
53       */
54      protected File testOutputDirectory;
55  
56      /**
57       * The encoding of source files.
58       *
59       * @parameter default-value="${project.build.sourceEncoding}"
60       */
61      protected String sourceEncoding;
62  
63      /**
64       * The Groovy compiler bytecode compatibility.  One of
65       * <ul>
66       *   <li>1.4</li>
67       *   <li>1.5</li>
68       *   <li>1.6</li>
69       *   <li>1.7</li>
70       *   <li>1.8</li>
71       * </ul>
72       *
73       * @parameter default-value="1.5"
74       */
75      protected String targetBytecode;
76  
77      /**
78       * Whether Groovy compiler should be set to debug.
79       *
80       * @parameter default-value="false"
81       */
82      protected boolean debug;
83  
84      /**
85       * Whether Groovy compiler should be set to verbose.
86       *
87       * @parameter default-value="false"
88       */
89      protected boolean verbose;
90  
91      /**
92       * Groovy compiler warning level.  Should be one of:
93       * <dl>
94       *   <dt>0</dt>
95       *     <dd>None</dd>
96       *   <dt>1</dt>
97       *     <dd>Likely Errors</dd>
98       *   <dt>2</dt>
99       *     <dd>Possible Errors</dd>
100      *   <dt>3</dt>
101      *     <dd>Paranoia</dd>
102      * </dl>
103      *
104      * @parameter default-value="1"
105      */
106     protected int warningLevel;
107 
108     /**
109      * Groovy compiler error tolerance
110      * (the number of non-fatal errors (per unit) that should be tolerated before compilation is aborted).
111      *
112      * @parameter default-value="0"
113      */
114     protected int tolerance;
115 
116     /**
117      * Allow setting whether to support invokeDynamic (requires Java 7 or greater).
118      *
119      * @parameter property="invokeDynamic" default-value="false"
120      */
121     protected boolean invokeDynamic;
122 
123     /**
124      * Performs compilation of compile mojos.
125      *
126      * @param sourcesToCompile The sources to compile
127      * @param classpath The classpath to use for compilation
128      * @param compileOutputDirectory The directory to write the compiled class files to
129      * @throws ClassNotFoundException When a class needed for compilation cannot be found
130      * @throws InstantiationException When a class needed for compilation cannot be instantiated
131      * @throws IllegalAccessException When a method needed for compilation cannot be accessed
132      * @throws InvocationTargetException When a reflection invocation needed for compilation cannot be completed
133      * @throws java.net.MalformedURLException When a classpath element provides a malformed URL
134      */
135     @SuppressWarnings("unchecked")
136     protected synchronized void doCompile(final Set<File> sourcesToCompile, final List classpath, final File compileOutputDirectory)
137             throws ClassNotFoundException, InstantiationException, IllegalAccessException, InvocationTargetException, MalformedURLException {
138         if (sourcesToCompile == null || sourcesToCompile.isEmpty()) {
139             getLog().info("No sources specified for compilation.  Skipping.");
140             return;
141         }
142 
143         // get classes we need with reflection
144         Class<?> compilerConfigurationClass = Class.forName("org.codehaus.groovy.control.CompilerConfiguration");
145         Class<?> compilationUnitClass = Class.forName("org.codehaus.groovy.control.CompilationUnit");
146         Class<?> groovyClassLoaderClass = Class.forName("groovy.lang.GroovyClassLoader");
147 
148         // set up compile options
149         Object compilerConfiguration = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(compilerConfigurationClass));
150         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setDebug", boolean.class), compilerConfiguration, debug);
151         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setVerbose", boolean.class), compilerConfiguration, verbose);
152         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setWarningLevel", int.class), compilerConfiguration, warningLevel);
153         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTolerance", int.class), compilerConfiguration, tolerance);
154         if (getGroovyVersion().compareTo(new Version(2, 1, 3)) >= 0) {
155             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetBytecode", String.class), compilerConfiguration, targetBytecode);
156         } else {
157             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetBytecode", String.class), compilerConfiguration, "1.5");
158         }
159         if (sourceEncoding != null) {
160             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setSourceEncoding", String.class), compilerConfiguration, sourceEncoding);
161         }
162         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "setTargetDirectory", String.class), compilerConfiguration, compileOutputDirectory.getAbsolutePath());
163         if (getGroovyVersion().compareTo(new Version(2, 0, 0, "beta-3")) >= 0 && invokeDynamic) {
164             if (isGroovyIndy()) {
165                 Map<java.lang.String, java.lang.Boolean> optimizationOptions = (Map<String, Boolean>) ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilerConfigurationClass, "getOptimizationOptions"), compilerConfiguration);
166                 optimizationOptions.put("indy", true);
167                 optimizationOptions.put("int", false);
168             } else {
169                 getLog().warn("Requested to use InvokeDynamic option but the version of Groovy on the project classpath doesn't support it.  Ignoring invokeDynamic option.");
170             }
171             if (!isJavaSupportIndy()) {
172                 throw new IllegalArgumentException("Tried to use to use an indy version of Groovy, but your Java version (" + getJavaVersionString() + ") doesn't support it.  Failing build.");
173             }
174         }
175 
176         // append project classpath to groovyClassLoader and transformLoader
177         Object groovyClassLoader = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyClassLoaderClass, ClassLoader.class, compilerConfigurationClass), compilationUnitClass.getClassLoader(), compilerConfiguration);
178         Object transformLoader = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(groovyClassLoaderClass, ClassLoader.class), compilationUnitClass.getClassLoader());
179         getLog().debug("Classpath: ");
180         if (classpath != null) {
181             for (Object classpathElement : classpath) {
182                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyClassLoaderClass, "addURL", URL.class), groovyClassLoader, new File((String) classpathElement).toURI().toURL());
183                 ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(groovyClassLoaderClass, "addURL", URL.class), transformLoader, new File((String) classpathElement).toURI().toURL());
184                 getLog().debug("    " + classpathElement);
185             }
186         }
187 
188         // add Groovy sources
189         Object compilationUnit;
190         if (getGroovyVersion().compareTo(new Version(1, 6, 0)) >= 0) {
191             compilationUnit = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(compilationUnitClass, compilerConfigurationClass, CodeSource.class, groovyClassLoaderClass, groovyClassLoaderClass), compilerConfiguration, null, groovyClassLoader, transformLoader);
192         } else {
193             compilationUnit = ReflectionUtils.invokeConstructor(ReflectionUtils.findConstructor(compilationUnitClass, compilerConfigurationClass, CodeSource.class, groovyClassLoaderClass), compilerConfiguration, null, groovyClassLoader);
194         }
195         getLog().debug("Adding Groovy to compile:");
196         for (File source : sourcesToCompile) {
197             getLog().debug("    " + source);
198             ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilationUnitClass, "addSource", File.class), compilationUnit, source);
199         }
200 
201         // compile the classes
202         ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilationUnitClass, "compile"), compilationUnit);
203 
204         // log compiled classes
205         List classes = (List) ReflectionUtils.invokeMethod(ReflectionUtils.findMethod(compilationUnitClass, "getClasses"), compilationUnit);
206         getLog().info("Compiled " + classes.size() + " file" + (classes.size() > 1 || classes.size() == 0 ? "s" : "") + ".");
207     }
208 
209     /**
210      * Determines whether this mojo can be run with the version of Groovy supplied.
211      * Must be >= 1.5.0 because not all the classes needed were available and
212      * functioning correctly in previous versions.
213      *
214      * @return <code>true</code> only if the version of Groovy supports this mojo.
215      */
216     protected boolean groovyVersionSupportsAction() {
217         return getGroovyVersion() != null && getGroovyVersion().compareTo(MIN_GROOVY_VERSION) >= 0;
218     }
