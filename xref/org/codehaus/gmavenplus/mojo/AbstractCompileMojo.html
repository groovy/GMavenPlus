304     /**
305      * Performs compilation of compile mojos.
306      *
307      * @param sources                the sources to compile
308      * @param classpath              the classpath to use for compilation
309      * @param compileOutputDirectory the directory to write the compiled class files to
310      * @throws ClassNotFoundException    when a class needed for compilation cannot be found
311      * @throws InstantiationException    when a class needed for compilation cannot be instantiated
312      * @throws IllegalAccessException    when a method needed for compilation cannot be accessed
313      * @throws InvocationTargetException when a reflection invocation needed for compilation cannot be completed
314      * @throws MalformedURLException     when a classpath element provides a malformed URL
315      */
316     @SuppressWarnings({"rawtypes"})
317     protected synchronized void doCompile(final Set<File> sources, final List classpath, final File compileOutputDirectory)
318             throws ClassNotFoundException, InstantiationException, IllegalAccessException, InvocationTargetException, MalformedURLException {
319         if (sources == null || sources.isEmpty()) {
320             getLog().info("No sources specified for compilation. Skipping.");
321             return;
322         }
323 
324         setupClassWrangler(classpath, includeClasspath);
325 
326         logPluginClasspath();
327         classWrangler.logGroovyVersion(mojoExecution.getMojoDescriptor().getGoal());
328 
329         if (groovyVersionSupportsAction() && !skipBytecodeCheck) {
330             verifyGroovyVersionSupportsTargetBytecode();
331         } else {
332             getLog().error("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support compilation. The minimum version of Groovy required is " + minGroovyVersion + ". Skipping compiling.");
333             return;
334         }
335 
336         // get classes we need with reflection
337         Class<?> compilerConfigurationClass = classWrangler.getClass("org.codehaus.groovy.control.CompilerConfiguration");
338         Class<?> compilationUnitClass = classWrangler.getClass("org.codehaus.groovy.control.CompilationUnit");
339         Class<?> groovyClassLoaderClass = classWrangler.getClass("groovy.lang.GroovyClassLoader");
340 
341         // setup compile options
342         Object compilerConfiguration = setupCompilerConfiguration(compileOutputDirectory, compilerConfigurationClass);
343         Object groovyClassLoader = invokeConstructor(findConstructor(groovyClassLoaderClass, ClassLoader.class, compilerConfigurationClass), classWrangler.getClassLoader(), compilerConfiguration);
344         Object transformLoader = invokeConstructor(findConstructor(groovyClassLoaderClass, ClassLoader.class), classWrangler.getClassLoader());
345 
346         // add Groovy sources
347         Object compilationUnit = setupCompilationUnit(sources, compilerConfigurationClass, compilationUnitClass, groovyClassLoaderClass, compilerConfiguration, groovyClassLoader, transformLoader);
348 
349         // compile the classes
350         invokeMethod(findMethod(compilationUnitClass, "compile"), compilationUnit);
351 
352         // log compiled classes
353         List classes = (List) invokeMethod(findMethod(compilationUnitClass, "getClasses"), compilationUnit);
354         getLog().info("Compiled " + classes.size() + " file" + (classes.size() != 1 ? "s" : "") + ".");
355     }

357     /**
358      * Sets up the CompilationUnit to use for compilation.
359      *
360      * @param sources                    the sources to compile
361      * @param compilerConfigurationClass the CompilerConfiguration class
362      * @param compilationUnitClass       the CompilationUnit class
363      * @param groovyClassLoaderClass     the GroovyClassLoader class
364      * @param compilerConfiguration      the CompilerConfiguration
365      * @param groovyClassLoader          the GroovyClassLoader
366      * @param transformLoader            the GroovyClassLoader to use for transformation
367      * @return the CompilationUnit
368      * @throws InstantiationException    when a class needed for setting up compilation unit cannot be instantiated
369      * @throws IllegalAccessException    when a method needed for setting up compilation unit cannot be accessed
370      * @throws InvocationTargetException when a reflection invocation needed for setting up compilation unit cannot be completed
371      */
372     protected Object setupCompilationUnit(final Set<File> sources, final Class<?> compilerConfigurationClass, final Class<?> compilationUnitClass, final Class<?> groovyClassLoaderClass, final Object compilerConfiguration, final Object groovyClassLoader, final Object transformLoader) throws InvocationTargetException, IllegalAccessException, InstantiationException {
373         Object compilationUnit;
374         if (groovyAtLeast(GROOVY_1_6_0)) {
375             compilationUnit = invokeConstructor(findConstructor(compilationUnitClass, compilerConfigurationClass, CodeSource.class, groovyClassLoaderClass, groovyClassLoaderClass), compilerConfiguration, null, groovyClassLoader, transformLoader);
376         } else {
377             compilationUnit = invokeConstructor(findConstructor(compilationUnitClass, compilerConfigurationClass, CodeSource.class, groovyClassLoaderClass), compilerConfiguration, null, groovyClassLoader);
378         }
379         getLog().debug("Adding Groovy to compile:");
380         Method addSourceMethod = findMethod(compilationUnitClass, "addSource", File.class);
381         for (File source : sources) {
382             getLog().debug("    " + source);
383             invokeMethod(addSourceMethod, compilationUnit, source);
384         }
385 
386         return compilationUnit;
387     }

389     /**
390      * Sets up the CompilationConfiguration to use for compilation.
391      *
392      * @param compileOutputDirectory     the directory to write the compiled classes to
393      * @param compilerConfigurationClass the CompilerConfiguration class
394      * @return the CompilerConfiguration
395      * @throws ClassNotFoundException    when a class needed for setting up CompilerConfiguration cannot be found
396      * @throws InstantiationException    when a class needed for setting up CompilerConfiguration cannot be instantiated
397      * @throws IllegalAccessException    when a method needed for setting up CompilerConfiguration cannot be accessed
398      * @throws InvocationTargetException when a reflection invocation needed for setting up CompilerConfiguration cannot be completed
399      */
400     @SuppressWarnings({"rawtypes", "unchecked"})
401     protected Object setupCompilerConfiguration(final File compileOutputDirectory, final Class<?> compilerConfigurationClass) throws InvocationTargetException, IllegalAccessException, InstantiationException, ClassNotFoundException {
402         Object compilerConfiguration = invokeConstructor(findConstructor(compilerConfigurationClass));
403         if (configScript != null) {
404             if (!configScript.exists()) {
405                 getLog().warn("Configuration script file (" + configScript.getAbsolutePath() + ") doesn't exist. Ignoring configScript parameter.");
406             } else if (groovyOlderThan(GROOVY_2_1_0_BETA1)) {
407                 getLog().warn("Requested to use configScript, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_2_1_0_BETA1 + " or newer). Ignoring configScript parameter.");
408             } else {
409                 Class<?> bindingClass = classWrangler.getClass("groovy.lang.Binding");
410                 Class<?> importCustomizerClass = classWrangler.getClass("org.codehaus.groovy.control.customizers.ImportCustomizer");
411                 Class<?> groovyShellClass = classWrangler.getClass("groovy.lang.GroovyShell");
412 
413                 Object binding = invokeConstructor(findConstructor(bindingClass));
414                 invokeMethod(findMethod(bindingClass, "setVariable", String.class, Object.class), binding, "configuration", compilerConfiguration);
415                 Object shellCompilerConfiguration = invokeConstructor(findConstructor(compilerConfigurationClass));
416                 Object importCustomizer = invokeConstructor(findConstructor(importCustomizerClass));
417                 invokeMethod(findMethod(importCustomizerClass, "addStaticStar", String.class), importCustomizer, "org.codehaus.groovy.control.customizers.builder.CompilerCustomizationBuilder");
418                 List compilationCustomizers = (List) invokeMethod(findMethod(compilerConfigurationClass, "getCompilationCustomizers"), shellCompilerConfiguration);
419                 compilationCustomizers.add(importCustomizer);
420                 Object shell = invokeConstructor(findConstructor(groovyShellClass, ClassLoader.class, bindingClass, compilerConfigurationClass), classWrangler.getClassLoader(), binding, shellCompilerConfiguration);
421                 getLog().debug("Using configuration script " + configScript + " for compilation.");
422                 invokeMethod(findMethod(groovyShellClass, "evaluate", File.class), shell, configScript);
423             }
424         }
425         invokeMethod(findMethod(compilerConfigurationClass, "setDebug", boolean.class), compilerConfiguration, debug);
426         invokeMethod(findMethod(compilerConfigurationClass, "setVerbose", boolean.class), compilerConfiguration, verbose);
427         invokeMethod(findMethod(compilerConfigurationClass, "setWarningLevel", int.class), compilerConfiguration, warningLevel);
428         invokeMethod(findMethod(compilerConfigurationClass, "setTolerance", int.class), compilerConfiguration, tolerance);
429         invokeMethod(findMethod(compilerConfigurationClass, "setTargetBytecode", String.class), compilerConfiguration, translateJavacTargetToTargetBytecode(targetBytecode));
430         if (previewFeatures) {
431             if (isJavaSupportPreviewFeatures()) {
432                 if (groovyOlderThan(GROOVY_2_5_7) || (groovyAtLeast(GROOVY_2_6_0_ALPHA1) && groovyOlderThan(GROOVY_3_0_0_BETA1))) {
433                     getLog().warn("Requested to use preview features, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_2_5_7 + "/" + GROOVY_3_0_0_BETA1 + " or newer. No 2.6 version is supported. Ignoring previewFeatures parameter.");
434                 } else {
435                     invokeMethod(findMethod(compilerConfigurationClass, "setPreviewFeatures", boolean.class), compilerConfiguration, previewFeatures);
436                 }
437             } else {
438                 getLog().warn("Requested to use to use preview features, but your Java version (" + getJavaVersionString() + ") doesn't support it. Ignoring previewFeatures parameter.");
439             }
440         }
441         if (sourceEncoding != null) {
442             invokeMethod(findMethod(compilerConfigurationClass, "setSourceEncoding", String.class), compilerConfiguration, sourceEncoding);
443         }
444         invokeMethod(findMethod(compilerConfigurationClass, "setTargetDirectory", String.class), compilerConfiguration, compileOutputDirectory.getAbsolutePath());
445         if (invokeDynamic || groovyAtLeast(GROOVY_4_0_0_ALPHA1)) {
446             if (groovyAtLeast(GROOVY_2_0_0_BETA3)) {
447                 if (classWrangler.isGroovyIndy()) {
448                     if (isJavaSupportIndy()) {
449                         Map<String, Boolean> optimizationOptions = (Map<String, Boolean>) invokeMethod(findMethod(compilerConfigurationClass, "getOptimizationOptions"), compilerConfiguration);
450                         optimizationOptions.put("indy", true);
451                         optimizationOptions.put("int", false);
452                         getLog().info("invokedynamic enabled.");
453                     } else {
454                         getLog().warn("Requested to use to use invokedynamic, but your Java version (" + getJavaVersionString() + ") doesn't support it. Ignoring invokeDynamic parameter.");
455                     }
456                 } else {
457                     getLog().warn("Requested to use invokedynamic, but your Groovy version doesn't support it (must use have indy classifier). Ignoring invokeDynamic parameter.");
458                 }
459             } else {
460                 getLog().warn("Requested to use invokeDynamic, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_2_0_0_BETA3 + " or newer). Ignoring invokeDynamic parameter.");
461             }
462         }
463         if (parameters) {
464             if (groovyAtLeast(GROOVY_2_5_0_ALPHA1)) {
465                 if (isJavaSupportParameters()) {
466                     invokeMethod(findMethod(compilerConfigurationClass, "setParameters", boolean.class), compilerConfiguration, parameters);
467                 } else {
468                     getLog().warn("Requested to use to use parameters, but your Java version (" + getJavaVersionString() + ") doesn't support it. Ignoring parameters parameter.");
469                 }
470             } else {
471                 getLog().warn("Requested to use parameters, but your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support it (must be " + GROOVY_2_5_0_ALPHA1 + " or newer). Ignoring parameters parameter.");
472             }
473         }
474         if (groovyAtLeast(GROOVY_3_0_5)) {
475             if ((parallelParsing == null && groovyAtLeast(GROOVY_4_0_0_ALPHA1)) || (parallelParsing != null && parallelParsing)) {
476                 Map<String, Boolean> optimizationOptions = (Map<String, Boolean>) invokeMethod(findMethod(compilerConfigurationClass, "getOptimizationOptions"), compilerConfiguration);
477                 optimizationOptions.put("parallelParse", true);
478                 getLog().info("Parallel parsing enabled.");
479             } else {
480                 getLog().info("Parallel parsing disabled.");
481             }
482         }
483 
484         return compilerConfiguration;
485     }

487     /**
488      * Throws an exception if targetBytecode is not supported with this version of Groovy. That is, when Groovy added
489      * the option to org.codehaus.groovy.control.CompilerConfiguration and used it in
490      * org.codehaus.groovy.classgen.asm.WriterController.
491      */
492     protected void verifyGroovyVersionSupportsTargetBytecode() {
493         if ("20".equals(targetBytecode)) {
494             if (groovyOlderThan(GROOVY_4_0_6)) {
495                 throw new IllegalArgumentException("Target bytecode " + targetBytecode + " requires Groovy " + GROOVY_4_0_6 + " or newer.");
496             }
497         } else if ("19".equals(targetBytecode)) {
498             if (groovyOlderThan(GROOVY_4_0_2)) {
499                 throw new IllegalArgumentException("Target bytecode " + targetBytecode + " requires Groovy " + GROOVY_4_0_2 + " or newer.");
500             }
501         } else if ("18".equals(targetBytecode)) {
502             if (groovyOlderThan(GROOVY_4_0_0_BETA1)) {
503                 throw new IllegalArgumentException("Target bytecode " + targetBytecode + " requires Groovy " + GROOVY_4_0_0_BETA1 + " or newer.");
504             }
505         } else if ("17".equals(targetBytecode)) {
506             if (groovyOlderThan(GROOVY_3_0_8) || (groovyAtLeast(GROOVY_4_0_0_ALPHA1) && groovyOlderThan(GROOVY_4_0_0_ALPHA3))) {
507                 throw new IllegalArgumentException("Target bytecode " + targetBytecode + " requires Groovy " + GROOVY_3_0_8 + "/" + GROOVY_4_0_0_ALPHA3 + " or newer.");
508             }
509         } else if ("16".equals(targetBytecode)) {
510             if (groovyOlderThan(GROOVY_3_0_6)) {
511                 throw new IllegalArgumentException("Target bytecode " + targetBytecode + " requires Groovy " + GROOVY_3_0_6 + " or newer.");
512             }
513         } else if ("15".equals(targetBytecode)) {
514             if (groovyOlderThan(GROOVY_3_0_3)) {
515                 throw new IllegalArgumentException("Target bytecode " + targetBytecode + " requires Groovy " + GROOVY_3_0_3 + " or newer.");
516             }
