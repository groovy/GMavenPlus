17  package org.codehaus.gmavenplus.mojo;
18  
19  import org.apache.maven.shared.model.fileset.FileSet;
20  import org.apache.maven.shared.model.fileset.util.FileSetManager;
21  
22  import java.io.File;
23  import java.util.*;
24  
25  
26  /**
27   * This mojo provides access to the Groovy sources (including stubs).
28   *
29   * @author Keegan Witt
30   */
31  public abstract class AbstractGroovyStubSourcesMojo extends AbstractGroovySourcesMojo {
32      /**
33       * The location for the compiled classes.
34       *
35       * @parameter default-value="${project.build.directory}/generated-sources/groovy-stubs/main"
36       */
37      protected File stubsOutputDirectory;
38  
39      /**
40       * The location for the compiled test classes.
41       *
42       * @parameter default-value="${project.build.directory}/generated-sources/groovy-stubs/test"
43       */
44      protected File testStubsOutputDirectory;
45  
46      /**
47       * Gets the set of files of the main stubs.
48       *
49       * @return The set of files of the main stubs
50       */
51      protected Set<File> getStubs() {
52          Set<File> files = new HashSet<File>();
53          FileSetManager fileSetManager = new FileSetManager(getLog());
54  
55          FileSet fileSet = new FileSet();
56          fileSet.setDirectory(stubsOutputDirectory.getAbsolutePath());
57          fileSet.setIncludes(Arrays.asList(JAVA_SOURCES_PATTERN));
58          for (String file : fileSetManager.getIncludedFiles(fileSet)) {
59              files.add(new File(stubsOutputDirectory, file));
60          }
61  
62          return files;
63      }
64  
65      /**
66       * Gets the set of files of the test stubs.
67       *
68       * @return The set of files of the test stubs
69       */
70      protected Set<File> getTestStubs() {
71          Set<File> files = new HashSet<File>();
72          FileSetManager fileSetManager = new FileSetManager(getLog());
73  
74          FileSet fileSet = new FileSet();
75          fileSet.setDirectory(testStubsOutputDirectory.getAbsolutePath());
76          fileSet.setIncludes(Arrays.asList(JAVA_SOURCES_PATTERN));
77          for (String file : fileSetManager.getIncludedFiles(fileSet)) {
78              files.add(new File(testStubsOutputDirectory, file));
79          }
80  
81          return files;
82      }
83  
84  }
