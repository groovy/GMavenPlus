32  /**
33   * Launches a Groovy shell bound to the current project.
34   * Note that this mojo requires Groovy >= 1.5.0.
35   * Note that it references the plugin ClassLoader to pull in dependencies
36   * Groovy didn't include (for things like Ant for AntBuilder, Ivy for @grab, and Jansi for Groovysh).
37   *
38   * @author Keegan Witt
39   * @since 1.1
40   */
41  @Mojo(name = "shell", requiresDependencyResolution = ResolutionScope.TEST, configurator = "include-project-test-dependencies")
42  public class ShellMojo extends AbstractToolsMojo {
43  
44      /**
45       * Groovy shell verbosity level. Should be one of:
46       * <ul>
47       *   <li>QUIET</li>
48       *   <li>INFO</li>
49       *   <li>DEBUG</li>
50       *   <li>VERBOSE</li>
51       * </ul>
52       */
53      @Parameter(defaultValue = "QUIET")
54      protected String verbosity;
55  
56      /**
57       * Executes this mojo.
58       *
59       * @throws MojoExecutionException If an unexpected problem occurs (causes a "BUILD ERROR" message to be displayed)
60       */
61      @Override
62      public void execute() throws MojoExecutionException {
63          classWrangler = setupClasswrangler();
64  
65          logPluginClasspath();
66          classWrangler.logGroovyVersion(mojoExecution.getMojoDescriptor().getGoal());
67  
68          try {
69              getLog().debug("Project test classpath:\n" + project.getTestClasspathElements());
70          } catch (DependencyResolutionRequiredException e) {
71              getLog().warn("Unable to log project test classpath", e);
72          }
73  
74          if (groovyVersionSupportsAction()) {
75              final SecurityManager sm = System.getSecurityManager();
76              try {
77                  if (!allowSystemExits) {
78                      System.setSecurityManager(new NoExitSecurityManager());
79                  }
80  
81                  // get classes we need with reflection
82                  Class<?> shellClass = classWrangler.getClass("org.codehaus.groovy.tools.shell.Groovysh");
83                  Class<?> bindingClass = classWrangler.getClass("groovy.lang.Binding");
84                  Class<?> ioClass = classWrangler.getClass("org.codehaus.groovy.tools.shell.IO");
85                  Class<?> verbosityClass = classWrangler.getClass("org.codehaus.groovy.tools.shell.IO$Verbosity");
86                  Class<?> loggerClass = classWrangler.getClass("org.codehaus.groovy.tools.shell.util.Logger");
87  
88                  // create shell to run
89                  Object shell = setupShell(shellClass, bindingClass, ioClass, verbosityClass, loggerClass);
90  
91                  // run the shell
92                  invokeMethod(findMethod(shellClass, "run", String.class), shell, (String) null);
93              } catch (ClassNotFoundException e) {
94                  throw new MojoExecutionException("Unable to get a Groovy class from classpath (" + e.getMessage() + "). Do you have Groovy as a compile dependency in your project or the plugin?", e);
95              } catch (InvocationTargetException e) {
96                  if (e.getCause() instanceof NoClassDefFoundError && e.getCause().getMessage() != null && e.getCause().getMessage().contains("jline")) {
97                      throw new MojoExecutionException("Unable to get a JLine class from classpath. This might be because of a JLine version mismatch. If you are using Groovy < 2.2.0-beta-1, make sure you include JLine 1.0 as a runtime dependency in your project or the plugin.", e);
98                  } else {
99                      throw new MojoExecutionException("Error occurred while calling a method on a Groovy class from classpath.", e);
100                 }
101             } catch (IllegalAccessException e) {
102                 throw new MojoExecutionException("Unable to access a method on a Groovy class from classpath.", e);
103             } catch (InstantiationException e) {
104                 throw new MojoExecutionException("Error occurred while instantiating a Groovy class from classpath.", e);
105             } finally {
106                 if (!allowSystemExits) {
107                     System.setSecurityManager(sm);
108                 }
109             }
110         } else {
111             getLog().error("Your Groovy version (" + classWrangler.getGroovyVersionString() + ") doesn't support running a shell. The minimum version of Groovy required is " + minGroovyVersion + ". Skipping shell startup.");
112         }
113     }
114 
115     /**
116      * Creates the Groovysh to run.
117      *
118      * @param shellClass the Groovysh class
119      * @param bindingClass the Binding class
120      * @param ioClass the IO class
121      * @param verbosityClass the Verbosity
122      * @param loggerClass the Logger class
123      * @return the Groovysh shell to run
124      * @throws InstantiationException when a class needed for setting up a shell cannot be instantiated
125      * @throws IllegalAccessException when a method needed for setting up a shell cannot be accessed
126      * @throws InvocationTargetException when a reflection invocation needed for setting up a shell cannot be completed
127      */
128     protected Object setupShell(final Class<?> shellClass, final Class<?> bindingClass, final Class<?> ioClass, final Class<?> verbosityClass, final Class<?> loggerClass) throws InvocationTargetException, IllegalAccessException, InstantiationException {
129         Object binding = invokeConstructor(findConstructor(bindingClass));
130         initializeProperties();
131         Method setVariable = findMethod(bindingClass, "setVariable", String.class, Object.class);
132         if (bindPropertiesToSeparateVariables) {
133             for (Object k : properties.keySet()) {
134                 invokeMethod(setVariable, binding, k, properties.get(k));
135             }
136         } else {
137             invokeMethod(setVariable, binding, "properties", properties);
138         }
139         Object io = invokeConstructor(findConstructor(ioClass));
140         invokeMethod(findMethod(ioClass, "setVerbosity", verbosityClass), io, invokeStaticMethod(findMethod(verbosityClass, "forName", String.class), verbosity));
141         findField(loggerClass, "io", ioClass).set(null, io);
142 
143         return invokeConstructor(findConstructor(shellClass, ClassLoader.class, bindingClass, ioClass), Thread.currentThread().getContextClassLoader(), binding, io);
144     }
