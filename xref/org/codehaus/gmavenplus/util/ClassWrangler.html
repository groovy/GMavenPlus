36   * Handles getting Groovy classes and version from the specified classpath.

44       * Cached Groovy version.

46      private String groovyVersion = null;

49       * Cached whether Groovy supports invokedynamic (indy jar).
50       */
51      private Boolean isIndy = null;

63      /**
64       * Creates a new ClassWrangler using the specified ClassLoader.
65       *
66       * @param classLoaderForLoading the ClassLoader to use to load classes
67       * @param pluginLog the Maven log to use for logging
68       */
69      public ClassWrangler(final ClassLoader classLoaderForLoading, final Log pluginLog) {
70          log = pluginLog;
71          classLoader = classLoaderForLoading;
72      }

74      /**
75       * Creates a new ClassWrangler using a new ClassLoader, loaded with the items from the specified classpath.
76       *
77       * @param classpath the classpath to load the new ClassLoader with
78       * @param pluginLog the Maven log to use for logging
79       * @throws MalformedURLException when a classpath element provides a malformed URL
80       */
81      public ClassWrangler(final List<?> classpath, final Log pluginLog) throws MalformedURLException {
82          log = pluginLog;
83          // create an isolated ClassLoader with all the appropriate project dependencies in it
84          classLoader = createNewClassLoader(classpath);
85          Thread.currentThread().setContextClassLoader(classLoader);
86      }

88      /**
89       * Gets the version string of Groovy used from classpath.
90       *
91       * @return The version string of Groovy used by the project
92       */
93      public String getGroovyVersionString() {
94          if (groovyVersion == null) {
95              // this method should work for all Groovy versions >= 1.6.6
96              try {
97                  Class<?> groovySystemClass = getClass("groovy.lang.GroovySystem");
98                  String ver = (String) invokeStaticMethod(findMethod(groovySystemClass, "getVersion"));
99                  if (ver != null && ver.length() > 0) {
100                     groovyVersion = ver;
101                 }
102             } catch (ClassNotFoundException e) {
103                 // do nothing, will try another way
104             } catch (IllegalAccessException e) {
105                 // do nothing, will try another way
106             } catch (InvocationTargetException e) {
107                 // do nothing, will try another way
108             } catch (IllegalArgumentException e) {
109                 // do nothing, will try another way
110             }
111 
112             // this should work for Groovy versions < 1.6.6 (technically can work up to 1.9.0)
113             if (groovyVersion == null) {
114                 log.info("Unable to get Groovy version from GroovySystem, trying InvokerHelper.");
115                 try {
116                     Class<?> invokerHelperClass = getClass("org.codehaus.groovy.runtime.InvokerHelper");
117                     String ver = (String) invokeStaticMethod(findMethod(invokerHelperClass, "getVersion"));
118                     if (ver != null && ver.length() > 0) {
119                         groovyVersion = ver;
120                     }
121                 } catch (ClassNotFoundException e) {
122                     // do nothing, will try another way
123                 } catch (IllegalAccessException e) {
124                     // do nothing, will try another way
125                 } catch (InvocationTargetException e) {
126                     // do nothing, will try another way
127                 } catch (IllegalArgumentException e) {
128                     // do nothing, will try another way
129                 }
130             }
131 
132             /*
133              * This handles the circumstances in which neither the GroovySystem or InvokerHelper methods
134              * worked (GAE with versions older than 1.6.6 is one example, see
135              * https://jira.codehaus.org/browse/GROOVY-3884). One case this can't handle properly is uber
136              * jars that include Groovy. It should also be noted this method assumes jars will be named
137              * in the Maven convention (<artifactId>-<version>-<classifier>.jar).
138              */
139             if (groovyVersion == null) {
140                 log.warn("Unable to get Groovy version from InvokerHelper or GroovySystem, trying jar name.");
141                 String jar = getGroovyJar();
142                 int idx = Integer.MAX_VALUE;
143                 for (int i = 0; i < 9; i++) {
144                     int newIdx = jar.indexOf("-" + i);
145                     if (newIdx >= 0 && newIdx < idx) {
146                         idx = newIdx;
147                     }
148                 }
149                 if (idx < Integer.MAX_VALUE) {
150                     groovyVersion = jar.substring(idx + 1, jar.length() - 4).replace("-indy", "").replace("-grooid", "");
151                 }
152             }
153         }
154 
155         return groovyVersion;
156     }

158     /**
159      * Gets the version of Groovy used from the classpath.
160      *
161      * @return The version of Groovy used by the project
162      */
163     public Version getGroovyVersion() {
164         try {
165             return Version.parseFromString(getGroovyVersionString());
166         } catch (Exception e) {
167             log.error("Unable to determine Groovy version. Is Groovy declared as a dependency?");
168             return null;
169         }
170     }

172     /**
173      * Determines whether the detected Groovy version is the specified version or newer.
174      *
175      * @param detectedVersion the detected Groovy version
176      * @param compareToVersion the version to compare the detected Groovy version to
177      * @return <code>true</code> if the detected Groovy version is the specified version or newer, <code>false</code> otherwise
178      */
179     public static boolean groovyAtLeast(Version detectedVersion, Version compareToVersion) {
180         return detectedVersion.compareTo(compareToVersion) >= 0;
181     }
182 
183     /**
184      * Determines whether the detected Groovy version is the specified version.
185      *
186      * @param detectedVersion the detected Groovy version
187      * @param compareToVersion the version to compare the detected Groovy version to
188      * @return <code>true</code> if the detected Groovy version is the specified version, <code>false</code> otherwise
189      */
190     public static boolean groovyIs(Version detectedVersion, Version compareToVersion) {
191         return detectedVersion.compareTo(compareToVersion) == 0;
192     }

195      * Determines whether the detected Groovy version is newer than the specified version.

216     /**
217      * Gets whether the version of Groovy on the classpath supports invokedynamic.
218      *
219      * @return <code>true</code> if the version of Groovy uses invokedynamic,
220      *         <code>false</code> if not or Groovy dependency cannot be found.
221      */
222     public boolean isGroovyIndy() {
223         if (isIndy == null) {
224             try {
225                 getClass("org.codehaus.groovy.vmplugin.v7.IndyInterface");
226                 isIndy = true;
227             } catch (ClassNotFoundException e) {
228                 isIndy = false;
229             }
230         }
231 
232         return isIndy;
233     }

235     /**
236      * Returns the filename of the Groovy jar on the classpath.
237      *
238      * @return the Groovy jar filename
239      */
240     public String getGroovyJar() {
241         try {
242             String groovyObjectClassPath = getJarPath();
243             String groovyJar = null;
244             if (groovyObjectClassPath != null) {
245                 groovyJar = groovyObjectClassPath.replaceAll("!.+", "");
246                 groovyJar = groovyJar.substring(groovyJar.lastIndexOf("/") + 1);
247             }
248 
249             return groovyJar;
250         } catch (ClassNotFoundException e) {
251             log.error("Unable to determine Groovy version. Is Groovy declared as a dependency?");
252             return null;
253         }
254     }

256     /**
257      * Returns the path of the Groovy jar on the classpath.
258      *
259      * @return the path of the Groovy jar
260      * @throws ClassNotFoundException when Groovy couldn't be found on the classpath
261      */
262     protected String getJarPath() throws ClassNotFoundException {
263         Class<?> groovyObjectClass = getClass("groovy.lang.GroovyObject");
264         String groovyObjectClassPath = String.valueOf(groovyObjectClass.getResource("/" + groovyObjectClass.getName().replace('.', '/') + ".class"));
265         if (groovyObjectClassPath == null) {
266             CodeSource codeSource = groovyObjectClass.getProtectionDomain().getCodeSource();
267             if (codeSource != null) {
268                 groovyObjectClassPath = String.valueOf(codeSource.getLocation());
269             }
270         }
271         return groovyObjectClassPath;
272     }

274     /**
275      * Logs the version of groovy used by this mojo.
276      *
277      * @param goal The goal to mention in the log statement showing Groovy version

279     public void logGroovyVersion(final String goal) {

285     /**
286      * Creates a new ClassLoader with the specified classpath.
287      *
288      * @param classpath the classpath (a list of file path Strings) to include in the new loader
289      * @return the new ClassLoader
290      * @throws MalformedURLException when a classpath element provides a malformed URL
291      */
292     public ClassLoader createNewClassLoader(final List<?> classpath) throws MalformedURLException {
293         List<URL> urlsList = new ArrayList<URL>();
294         for (Object classPathObject : classpath) {
295             String path = (String) classPathObject;
296             urlsList.add(new File(path).toURI().toURL());
297         }
298         URL[] urlsArray = urlsList.toArray(new URL[0]);
299         return new URLClassLoader(urlsArray, ClassLoader.getSystemClassLoader());
300     }

302     /**
303      * Gets a class for the given class name.
304      *
305      * @param className the class name to retrieve the class for
306      * @return the class for the given class name
307      * @throws ClassNotFoundException when a class for the specified class name cannot be found
308      */
309     public Class<?> getClass(final String className) throws ClassNotFoundException {
310         return Class.forName(className, true, classLoader);
311     }
