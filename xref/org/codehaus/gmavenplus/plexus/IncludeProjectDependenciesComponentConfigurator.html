28   * A custom ComponentConfigurator which adds the project's compile-time classpath elements.
29   *
30   * @author Brian Jackson

33   * @plexus.component role="org.codehaus.plexus.component.configurator.ComponentConfigurator"
34   *                   role-hint="include-project-dependencies"

36   *                     role-hint="default"
37   */
38  @SuppressWarnings("deprecation")
39  public class IncludeProjectDependenciesComponentConfigurator extends AbstractComponentConfigurator {

41  
42      public void configureComponent(Object component, PlexusConfiguration configuration, ExpressionEvaluator expressionEvaluator,
43                                     ClassRealm containerRealm, ConfigurationListener listener) throws ComponentConfigurationException {
44          addProjectCompileDependenciesToClassRealm(expressionEvaluator, containerRealm);
45          converterLookup.registerConverter(new ClassRealmConverter(containerRealm));
46          ObjectWithFieldsConverter converter = new ObjectWithFieldsConverter();
47  //        converter.processConfiguration(converterLookup, component, containerRealm.getParentClassLoader(), configuration, expressionEvaluator, listener);
48          converter.processConfiguration(converterLookup, component, containerRealm.getClassLoader(), configuration, expressionEvaluator, listener);
49      }

51      /**
52       * Adds the project's compile dependencies to the specified ClassRealm.
53       *
54       * @param expressionEvaluator the expression evaluator to use to get project elements
55       * @param containerRealm the ClassRealm to add dependencies to
56       * @throws ComponentConfigurationException
57       */
58      private void addProjectCompileDependenciesToClassRealm(ExpressionEvaluator expressionEvaluator, ClassRealm containerRealm) throws ComponentConfigurationException {
59          List classpathElements;
60  
61          try {
62              classpathElements = (List) expressionEvaluator.evaluate("${project.compileClasspathElements}");
63          } catch (ExpressionEvaluationException e) {
64              throw new ComponentConfigurationException("There was a problem evaluating: ${project.compileClasspathElements}.", e);
65          }
66  
67          // Add the project dependencies to the ClassRealm
68          final URL[] urls = buildURLs(classpathElements);
69          for (URL url : urls) {
70  //            containerRealm.addURL(url);
71              containerRealm.addConstituent(url);
72          }
73      }
74  
75      /**
76       * Create an array of URLs for all the elements in the classpath.
77       *
78       * @param classpathElements the classpath elements to create URLs for
79       * @return URLs for all the classpath elements
80       * @throws ComponentConfigurationException
81       */
82      private URL[] buildURLs(List classpathElements) throws ComponentConfigurationException {
83          List<URL> urls = new ArrayList<URL>(classpathElements.size());
84          for (Object element : classpathElements) {
85              try {
86                  final URL url = new File((String) element).toURI().toURL();
87                  urls.add(url);
88                  // commented out because debug seems to be on all the time
89  //                LOG.debug("Added to project class loader: " + url);
90              } catch (MalformedURLException e) {
91                  throw new ComponentConfigurationException("Unable to access project dependency: " + element + ".", e);
92              }
93          }
94  
95          return urls.toArray(new URL[urls.size()]);
96      }
