48      @Test
49      public void testConfigureComponent_1() throws Exception {
50          ConverterLookup converterLookup = Mockito.mock(ConverterLookup.class);
51          Field modifiersField = configurator.getClass().getSuperclass().getSuperclass().getSuperclass().getDeclaredField("converterLookup");
52          modifiersField.setAccessible(true);
53          modifiersField.set(configurator, converterLookup);
54          Object component = Mockito.mock(Object.class);
55          PlexusConfiguration configuration = Mockito.mock(PlexusConfiguration.class);
56          ExpressionEvaluator expressionEvaluator = Mockito.mock(ExpressionEvaluator.class);
57          org.codehaus.plexus.classworlds.realm.ClassRealm containerRealm = Mockito.mock(org.codehaus.plexus.classworlds.realm.ClassRealm.class);
58          ConfigurationListener listener = Mockito.mock(ConfigurationListener.class);
59          Mockito.doNothing().when(configurator).addDependenciesToClassRealm(Mockito.any(ExpressionEvaluator.class), Mockito.any(AbstractIncludeProjectDependenciesComponentConfigurator.Classpath.class), Mockito.any(org.codehaus.plexus.classworlds.realm.ClassRealm.class));
60  
61          configurator.configureComponent(component, configuration, expressionEvaluator, containerRealm, listener);
62  
63          Mockito.verify(configurator, Mockito.atLeastOnce()).addDependenciesToClassRealm(Mockito.any(ExpressionEvaluator.class), Mockito.any(AbstractIncludeProjectDependenciesComponentConfigurator.Classpath.class), Mockito.any(org.codehaus.plexus.classworlds.realm.ClassRealm.class));
64          Mockito.verify(converterLookup, Mockito.atLeastOnce()).registerConverter(Mockito.any(ClassRealmConverter.class));
65      }

86      @Test
87      public void testAddProjectRuntimeDependenciesToClassRealm() throws Exception {
88          ExpressionEvaluator expressionEvaluator = Mockito.mock(ExpressionEvaluator.class);
89          List classpathElements = Arrays.asList("CLASSPATH_ELEMENT");
90          Mockito.doReturn(classpathElements).when(expressionEvaluator).evaluate(Mockito.anyString());
91          org.codehaus.classworlds.ClassRealm containerRealm = Mockito.mock(org.codehaus.classworlds.ClassRealm.class);
92          configurator.addDependenciesToClassRealm(expressionEvaluator, IncludeProjectRuntimeDependenciesComponentConfigurator.Classpath.RUNTIME, containerRealm);
93          Mockito.verify(expressionEvaluator, Mockito.times(1)).evaluate(Mockito.anyString());
94          Mockito.verify(containerRealm, Mockito.times(1)).addConstituent(Mockito.any(URL.class));
95      }
